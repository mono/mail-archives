text = lookup_widget(widget, "myField");
if (input_invalid(text)) {
  dialog1 = create_dialog1();
  gtk_dialog_run (GTK_DIALOG(dialog1));
  gtk_widget_hide(dialog1);
  gtk_widget_grab_focus(text);
}
return FALSE;
