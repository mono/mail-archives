And if you use Glade, must call gtk_window_fullscreen after get object
from interface:

/* gcc window.c -o window `pkg-config --cflags --libs gtk+-2.0` */

#include <gtk/gtk.h>

const gchar *interface = "<interface><object class=\"GtkWindow\" id=
\"window\"></object></interface>";

int main(int argc, char *argv[]) {
  gtk_init(&argc, &argv);
  GtkBuilder *builder = gtk_builder_new();
  gtk_builder_add_from_string(builder, interface, -1, NULL);
  GtkWidget *window = GTK_WIDGET(gtk_builder_get_object(builder,
"window"));
  gtk_window_fullscreen(GTK_WINDOW(window));
  g_signal_connect(G_OBJECT(window), "delete-event",
G_CALLBACK(gtk_main_quit), NULL);
  gtk_widget_show(window);
  gtk_main();
}
