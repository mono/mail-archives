--- Example ---

gboolean
on_clist1_button_press_event      (GtkWidget       *widget,
                                   GdkEventButton  *event,
                                   gpointer user_data)
{

if (event->type =3D=3D GDK_2BUTTON_PRESS)
   {
      /* A double click occured */

      /* Your stuff goes here */
=20

   }
  return FALSE;
}
