And if you use Glade, must call gtk_window_fullscreen after get object from interface:<BR>
<BR>
/* gcc window.c -o window `pkg-config --cflags --libs gtk+-2.0` */<BR>
<BR>
#include &lt;gtk/gtk.h&gt;<BR>
<BR>
const gchar *interface = &quot;&lt;interface&gt;&lt;object class=\&quot;GtkWindow\&quot; id=\&quot;window\&quot;&gt;&lt;/object&gt;&lt;/interface&gt;&quot;;<BR>
<BR>
int main(int argc, char *argv[]) {<BR>
&nbsp; gtk_init(&amp;argc, &amp;argv);<BR>
&nbsp; GtkBuilder *builder = gtk_builder_new();<BR>
&nbsp; gtk_builder_add_from_string(builder, interface, -1, NULL);<BR>
<B>&nbsp; GtkWidget *window = GTK_WIDGET(gtk_builder_get_object(builder, &quot;window&quot;));</B><BR>
<B>&nbsp; gtk_window_fullscreen(GTK_WINDOW(window));</B><BR>
&nbsp; g_signal_connect(G_OBJECT(window), &quot;delete-event&quot;, G_CALLBACK(gtk_main_quit), NULL);<BR>
&nbsp; gtk_widget_show(window);<BR>
&nbsp; gtk_main();<BR>
}<BR>
