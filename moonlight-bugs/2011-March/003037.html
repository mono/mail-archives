Description of Problem:

Steps to reproduce the problem:
1. Enter website: http://www.fku-aura.com/applicationeng.html?offer!en
2. Scroll page down to Silverlight application

Actual Results:
You can see application in Silverlight with simply carousel. Below is
navigation panel and there are arrows. Pressing them will have NO effect.
However, just switch the application in Full Screen Mode (by simply clicking on
the background) and arrows will be changing carousel's items position
In my ubuntu firefox there are also several performance problems (in Windows
are not).

Expected Results:
Application should works fine in normal/full screen mode - in Windows
everything is ok.

How often does this happen? 
Always

Additional Information:
Application does not use DoubleAnimation objects to change carousel position
values, but DispatcherTimer (I know - wrong solution). This may be the reason
for poor performance in Moonlight (not in Silverlight)

This file can be useful. Arrows on navigation panel are DoubleImage objects (my
own class). DoubleImage in Moonlight works only in full screen mode

###############################################
using System;

using System.Net;

using System.Windows;

using System.Windows.Controls;

using System.Windows.Documents;

using System.Windows.Ink;

using System.Windows.Input;

using System.Windows.Media;

using System.Windows.Media.Animation;

using System.Windows.Shapes;



namespace MatrixReliability.SilverlightLibrary

{

    public class DoubleImage : UserControl

    {

        internal Image _image1; //only in use of emergency

        internal Image _image2; //only in use of emergency



        private Canvas _imagesOwnerCanvas;

        public Canvas ImagesOwnerCanvas

        {

            get { return _imagesOwnerCanvas; }

        }



        bool DoChange = true;



        public Image Image1

        {

            get { return _image1; }

        }



        public Image Image2

        {

            get { return _image2; }

        }



        public void Show()

        {

            Image1.Visibility = Visibility.Visible;

            Image2.Visibility = Visibility.Collapsed;

        }



        public void Hide()

        {

            Image1.Visibility = Visibility.Collapsed;

            Image2.Visibility = Visibility.Collapsed;

        }



        public void SetStartingVisibility()

        {

            _image2.Visibility = Visibility.Collapsed;

            _image1.Visibility = Visibility.Visible;

        }



        public bool Enabled

        {

            get { return _image1.IsHitTestVisible; }

            set

            {

                _image1.IsHitTestVisible = value;

                _image2.IsHitTestVisible = value;

            }

        }



        public void SetTag(string iTag)

        {

            _image1.Tag = iTag;

            _image2.Tag = iTag;

        }



        public DoubleImage()

        {

            _image1 = new Image();

            _image2 = new Image();

            _image1.MouseMove += new
MouseEventHandler(_image1onlyForRefParameter_MouseMove);

            _image2.MouseLeave += new
MouseEventHandler(_image2onlyForRefParameter_MouseLeave);

            SetStartingVisibility();

        }



        void _image2onlyForRefParameter_MouseLeave(object sender,
MouseEventArgs e)

        {

            if (!DoChange)

                return;

            ShowFirstImage();

        }



        void _image1onlyForRefParameter_MouseMove(object sender, MouseEventArgs
e)

        {

            if (!DoChange)

                return;

            ShowSecondImage();

        }



        public void ShowFirstImage()

        {

            _image2.Visibility = Visibility.Collapsed;

            _image1.Visibility = Visibility.Visible;

        }



        public void ShowSecondImage()

        {

            _image1.Visibility = Visibility.Collapsed;

            _image2.Visibility = Visibility.Visible;

        }



        public DoubleImage(string iImage1Source, string iImage2Source, Size
iImage1Size, Size iImage2Size)

        {

            _image1 = new Image();

            _image2 = new Image();

            _image1.MouseMove += new
MouseEventHandler(_image1onlyForRefParameter_MouseMove);

            _image2.MouseLeave += new
MouseEventHandler(_image2onlyForRefParameter_MouseLeave);

            ImageHandler.LoadFile(ref _image1, iImage1Source);

            _image1.Width = iImage1Size.Width;

            _image1.Height = iImage1Size.Height;

            ImageHandler.LoadFile(ref _image2, iImage2Source);

            _image2.Width = iImage2Size.Width;

            _image2.Height = iImage2Size.Height;

            SetStartingVisibility();

        }



        public DoubleImage(Size iImage1Size, Size iImage2Size)

        {

            _image1 = new Image();

            _image2 = new Image();

            _image1.MouseMove += new
MouseEventHandler(_image1onlyForRefParameter_MouseMove);

            _image2.MouseLeave += new
MouseEventHandler(_image2onlyForRefParameter_MouseLeave);

            _image1.Width = iImage1Size.Width;

            _image1.Height = iImage1Size.Height;

            _image2.Width = iImage2Size.Width;

            _image2.Height = iImage2Size.Height;

            SetStartingVisibility();

        }



        public void LoadImage(string iImageSource, string iImageSource2)

        {

            ImageHandler.LoadFile(ref _image1, iImageSource);

            ImageHandler.LoadFile(ref _image2, iImageSource2);

        }



        public void LoadImage1(string iImageSource)

        {

            ImageHandler.LoadFile(ref _image1, iImageSource);

        }



        public void LoadImage2(string iImageSource)

        {

            ImageHandler.LoadFile(ref _image2, iImageSource);

        }



        public void SetImage1Stretch(Stretch iStretch)

        {

            _image1.Stretch = iStretch;

        }



        public void SetImage2Stretch(Stretch iStretch)

        {

            _image2.Stretch = iStretch;

        }





        public double Left

        {

            get { return Canvas.GetLeft(_image1); }

            set 

            {

                _image1.SetValue(Canvas.LeftProperty, value);

                _image2.SetValue(Canvas.LeftProperty, value);

            }

        }



        public double Top

        {

            get { return Canvas.GetTop(_image1); }

            set

            {

                _image1.SetValue(Canvas.TopProperty, value);

                _image2.SetValue(Canvas.TopProperty, value);

            }

        }



        public void SetLeft(double iLeft)

        {

            _image1.SetValue(Canvas.LeftProperty, iLeft);

            _image2.SetValue(Canvas.LeftProperty, iLeft);

        }



        public void SetTop(double iTop)

        {

            _image1.SetValue(Canvas.TopProperty, iTop);

            _image2.SetValue(Canvas.TopProperty, iTop);

        }



        new public double Width

        {

            get { return _image1.Width; }

            set

            {

                if (value < 0)

                    value = 0;

                _image1.Width = value;

                _image2.Width = value;

            }

        }



        new public double Height

        {

            get { return _image1.Height; }

            set

            {

                if (value < 0)

                    value = 0;

                _image1.Height = value;

                _image2.Height = value;

            }

        }



        public void AddImagesItemsToCanvas(Canvas iCanvas)

        {

            if ((Image1.Parent != null)||(Image2.Parent != null))

                return;

            iCanvas.Children.Add(Image1);

            iCanvas.Children.Add(Image2);

            _imagesOwnerCanvas = iCanvas;

        }



        public void RemoveDoubleImageFromCanvas()

        {

            _imagesOwnerCanvas.Children.Remove(Image1);

            _imagesOwnerCanvas.Children.Remove(Image2);

        }



        public void SetDoubleImageChange(bool iDoDoubleChange)

        {

            DoChange = iDoDoubleChange;

        }



        public void AddMouseLeftButtonDown(MouseButtonEventHandler
iMouseButtonEventHandler)

        {

            _image2.MouseLeftButtonDown += iMouseButtonEventHandler;

        }



        public void SetName(string iName)

        {

            Name = iName;

        }

    }

}
