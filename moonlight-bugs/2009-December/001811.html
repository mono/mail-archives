When having a shape (eg. polygon) with a StrokeDashArray, when it's set to null
to remove the dashes nothing happens. Here's a simple example to demonstrate

-- Page.xaml
<UserControl xmlns="http://schemas.microsoft.com/winfx/2006/xaml/presentation" 
             xmlns:x="http://schemas.microsoft.com/winfx/2006/xaml" 
             Width="500" Height="350"
             x:Class="StrokeDashTest.Page"
>
    <StackPanel>
        <Polygon Fill="Red" Stroke="Blue" StrokeThickness="3"
            StrokeDashArray="2 2" Points="10,70 70,70 70,10"
            x:Name="dashedPolygon" />
        <Button Content="Remove Dashes" x:Name="btnRemoveDashes"
            Click="btnRemoveDashes_Click" />
    </StackPanel>
</UserControl>

-- Page.cs
using System;
using System.Collections.Generic;
using System.Windows;
using System.Windows.Controls;

namespace StrokeDashTest
{
    public partial class Page : UserControl
    {

        public Page ()
        {
            this.InitializeComponent ();
        }

        void btnRemoveDashes_Click (object sender, RoutedEventArgs e)
        {
            dashedPolygon.StrokeDashArray = null;
        }
    }
}


An easy workaround, though not pretty is:

dashedPolygon.StrokeThickness++;
dashedPolygon.StrokeThickness--;
