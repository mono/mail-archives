repro

Moonlight does not raise Canvas.Loaded event where the Canvas and the event are
set during Application.Startup event.

I'm attaching an entire repro app, but the app basically looks like:

using System;
using System.Collections.Generic;
using System.Windows;
using System.Windows.Controls;

namespace moonlight1
{
    public partial class App : ProcessingApplication
    {
        public App ()
            : base (new Action (Run))
        {
        }

        static void Run ()
        {
            System.Windows.Browser.HtmlPage.Window.Alert ("OK");
        }
    }

    public partial class ProcessingApplication : Application
    {
        public static ProcessingApplication Current { get; set; }

        public ProcessingApplication (Action run)
        {
            Current = this;
            Startup += delegate {
                var c = new Canvas ();
                this.RootVisual = c;
                c.Loaded += delegate { run (); };
            };
        }
    }
}

The repro is also available at
http://veritas-vos-liberabit.com/tmp/2009/moonlight1/bin/Debug/TestPage.html

On Silverlight it raises Javascript Alert.
