not, mono.png.

<html xmlns="http://www.w3.org/1999/xhtml" xml:lang="en">
<head>
<title>createFromXaml does not always download</title>
<meta></meta>
</head>
<body>

<script type="text/javascript">
function OnLoaded (sender, eventArgs)
{
        var control = document.getElementById ("agControl");
        var content = control.content;

        var xaml = "<Image Source='mono.png'/>";
        var image = content.createFromXaml (xaml);
}
</script>

<object type="application/x-silverlight" data="data:," id="agControl"
width="100%" height="100%">
<param name="background" value="#FFFFFF"/>
<param name="source" value="#xamlContent"/>
</object>

<script type="text/xaml" id="xamlContent">
<?xml version="1.0"?>
<Canvas xmlns="http://schemas.microsoft.com/winfx/2006/xaml/presentation"
        xmlns:x="http://schemas.microsoft.com/winfx/2006/xaml"
