using System;
using System.Collections;
using System.Collections.Generic;
using System.Xml;
using System.Xml.Linq;
using System.Linq;
using Moonlight.Gtk;

using System.Windows;
using System.Windows.Controls;
using System.Windows.Media;
using Moonlight;
using System.Windows.Shapes;

namespace MoonlightGtk
{
    class MainClass
    {

        const int width = 1024;
        const int height = 768;

        public static void Main (string[] args)
        {

            Gtk.Application.Init ();
            MoonlightRuntime.Init ();

            Gtk.Window w = new Gtk.Window ("Foo");

            w.DeleteEvent += delegate { Gtk.Application.Quit (); };

            w.SetSizeRequest (width, height);
            MoonlightHost h = new MoonlightHost ();
            
            Canvas c = new Canvas ();
            c.Background = new SolidColorBrush (Colors.Green);
            c.Width = width;
            c.Height = height;

            Grid g = new Grid ();

            Button b = new Button ();
            /*b.Width = width;
            b.Height = 100;
            b.Background = new SolidColorBrush (Colors.Blue);
            b.Content = "Click me";
            b.VerticalAlignment = VerticalAlignment.Top;
            */

            g.Children.Add (b);

            // creates a Circle

            Ellipse circle = new Ellipse ();
            circle.Width = 200;
            circle.Height = 200;
            circle.Fill = new SolidColorBrush (Colors.Yellow);
            circle.Stroke = new SolidColorBrush (Colors.Black);
            circle.StrokeThickness = 4;

            g.Children.Add (circle);

            c.Children.Add (g);
            
            h.Application.RootVisual = c;


            w.Add (h);
            w.ShowAll ();

            
            Gtk.Application.Run ();
        }
    }
}
