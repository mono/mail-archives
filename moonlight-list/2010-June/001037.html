> 
> Here is a sample build script
> 
> #!/bin/sh
> rm ./obj/*
> xamlg -sl2app:Example Main.xaml,Main.xaml.g.cs 
> 
> mv *.xaml.g.cs ./obj/
> 
> respack ./obj/Example.g.resources Main.xaml,Main.xaml
> 
> gmcs *.cs ./obj/*.xaml.g.cs \
>      -r:./References/Moonlight/System.Windows.dll \
>      -r:./References/Moonlight/System.Windows.Controls.dll \
>      -r:./References/Moonlight/System.Windows.Controls.Data.dll \
>      -r:./References/Moonlight/System.Windows.Browser.dll \
>      -target:library \
>      -out:./bin/Example.dll \
>      -resource:./obj/Example.g.resources -debug+
> 
> 
> It will build an Assembly called Example.dll which contains a Main.xaml. You
> will need to alter the moonlight assembly references (i copy the desktop
> dll's to a references folder in my project to make it easy to find).

TARGET=MyControl
XAML=Fred.xaml Foo.xaml
RESFILE=$(TARGET).g.resources
SRCS=Fred.xaml.cs Foo.xaml.cs
XAMLG=$(addsuffix .g.cs, $(XAML))

CSCFLAGS=-pkg:moonlight-desktop-2.0 
CSC=gmcs

$(TARGET).dll: $(XAMLG) $(SRCS)
	$(CSC) /out:$@ /target:library /resource:$(RESFILE) $^ $(CSCFLAGS)

$(RESFILE): $(XAMLG)

%.xaml.g.cs: %.xaml
	xamlg $<
	respack $(RESFILE) $<,$<

clean:
	@rm -f *.g.cs
	@rm -f $(RESFILE)
