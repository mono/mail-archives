TARGET=foo
XAML=Page.xaml App.xaml
SRCS=Page.xaml.cs App.xaml.cs
RESFILE=$(TARGET).g.resources
XAMLG=$(addsuffix .g.cs, $(XAML))
RESLIST:=
REFS="/r:System.dll" \
"/r:mscorlib.dll" \
"/r:System.Core.dll" \
"/r:System.Xml.dll"

CSCFLAGS=$(REFS) -pkg:moonlight-gtk-2.0,moonlight-desktop-2.0,dotnet35 /nologo /codepage:utf8 /optimize- /debug:+
CSC=gmcs

all: clean $(TARGET).xap
	mkdir -p bin/Desktop
	mv $(TARGET).xap bin/Desktop
	mv $(TARGET).dll bin/Desktop

$(TARGET).dll: $(XAMLG) $(SRCS) $(RESFILE)
	$(CSC) /out:$(TARGET).dll /target:library /res:$(RESFILE),$(TARGET).g.resources $(SRCS) $(XAMLG) $(CSCFLAGS)

$(TARGET).xap: $(TARGET).dll
	cp DesktopManifest.xaml AppManifest.xaml
	zip $(TARGET).xap $(TARGET).dll AppManifest.xaml
	rm AppManifest.xaml

$(RESFILE): $(XAML)
	respack $(RESFILE) $(XAML)

%.xaml.g.cs: %.xaml
	xamlg -sl2app:$(TARGET) $<,$<.g.cs

clean:
	@rm -f *.g.cs
	@rm -f $(RESFILE)
