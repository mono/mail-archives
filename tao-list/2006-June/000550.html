> BTW, what is demo.cs? Could you post that as well?

CartouhoGL's example from a few days ago:

> using System;
> using System.Collections;
> using Tao.Sdl;
>
> public class Video {
> 	public static void Main() {
> 		Sdl.SDL_Init(Sdl.SDL_INIT_VIDEO);
> 		int flags = (Sdl.SDL_ANYFORMAT);
> 		Sdl.SDL_SetVideoMode(320,200,16,flags);
> 		
> 		Sdl.SDL_Event evt;
> 		bool exit=false;
> 		
> 		while(!exit) {
> 		  Sdl.SDL_WaitEvent(out evt);
> 		  if (evt.type == Sdl.SDL_QUIT)
> 			exit = true;
> 		  if (evt.type == Sdl.SDL_MOUSEMOTION)
> 		  	Console.WriteLine("mouse");
> 		}
> 		Sdl.SDL_Quit();
> 	}
> }
