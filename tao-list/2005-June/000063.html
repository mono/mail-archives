int imageId;
int glId;
string file;

Ilut.ilutRenderer(Ilut.ILUT_OPENGL);

Il.ilBindImage(imageId);

if(!Il.ilLoadImage(file)) {
  Console.WriteLine("Could not open file, {0}, exiting.", file);
} else {
    Console.WriteLine("Width: {0} Height: {1}, Depth: {2}, Bpp: {3}",
    Il.ilGetInteger(Il.IL_IMAGE_WIDTH),
    Il.ilGetInteger(Il.IL_IMAGE_HEIGHT),
    Il.ilGetInteger(Il.IL_IMAGE_DEPTH),
    Il.ilGetInteger(Il.IL_IMAGE_BITS_PER_PIXEL));

    glId = Ilut.ilutGLBindTexImage();
}
