Hi Siva, I have some code that uses vertex buffer objects that's
working for me. You'll have to ignore some of the code specific to my
data structures. I have three integers in my class, vertexVboName,
normalVboName, and edgeVboName. I'm rendering the edges of the mesh on
top of the regular Phong-shaded rendering of it with edgeVboName, you
can remove that if it's not useful to you. To initialize the VBOs
(after loading GL_ARB_vertex_buffer_object), I have the following
method:
------
private void InitializeVertexBuffers()
{
    // Generate arrays with the mesh vertices and normals to pass to the GPU
    int numVertices = mesh.Faces.Count * 3;
    float[] vertexCache = new float[numVertices * 3];
    float[] normalCache = new float[numVertices * 3];
    int vertexCounter = 0;
    foreach (TriMesh.Face f in mesh.Faces)
    {
        foreach (TriMesh.Halfedge h in f.Halfedges)
        {
            TriMesh.Vertex v = h.ToVertex;

            vertexCache[vertexCounter] = v.Traits.Position.X;
            vertexCache[vertexCounter + 1] = v.Traits.Position.Y;
            vertexCache[vertexCounter + 2] = v.Traits.Position.Z;

            normalCache[vertexCounter] = h.Traits.Normal.X;
            normalCache[vertexCounter + 1] = h.Traits.Normal.Y;
            normalCache[vertexCounter + 2] = h.Traits.Normal.Z;

            vertexCounter += 3;
        }
    }

    // Generate an array with the edge vertices to pass to the GPU
    numVertices = mesh.Edges.Count * 2;
    float[] edgeCache = new float[numVertices * 3];
    vertexCounter = 0;
    foreach (TriMesh.Edge e in mesh.Edges)
    {
        Vector3F edgeStart = e.Vertex0.Traits.Position;
        Vector3F edgeEnd = e.Vertex1.Traits.Position;

        edgeCache[vertexCounter] = edgeStart.X;
        edgeCache[vertexCounter + 1] = edgeStart.Y;
        edgeCache[vertexCounter + 2] = edgeStart.Z;
        edgeCache[vertexCounter + 3] = edgeEnd.X;
        edgeCache[vertexCounter + 4] = edgeEnd.Y;
        edgeCache[vertexCounter + 5] = edgeEnd.Z;

        vertexCounter += 6;
    }

    // Pass vertices to the GPU
    if (Gl.glIsBufferARB(vertexVboName) == 0)   // New lists
    {
        Gl.glGenBuffersARB(1, out vertexVboName);
        Gl.glGenBuffersARB(1, out normalVboName);
        Gl.glGenBuffersARB(1, out edgeVboName);

        Gl.glBindBufferARB(Gl.GL_ARRAY_BUFFER_ARB, vertexVboName);
        Gl.glBufferDataARB(Gl.GL_ARRAY_BUFFER_ARB, numVertices * 3 *
sizeof(float), vertexCache, Gl.GL_DYNAMIC_DRAW_ARB);

        Gl.glBindBufferARB(Gl.GL_ARRAY_BUFFER_ARB, normalVboName);
        Gl.glBufferDataARB(Gl.GL_ARRAY_BUFFER_ARB, numVertices * 3 *
sizeof(float), normalCache, Gl.GL_DYNAMIC_DRAW_ARB);

        Gl.glBindBufferARB(Gl.GL_ARRAY_BUFFER_ARB, edgeVboName);
        Gl.glBufferDataARB(Gl.GL_ARRAY_BUFFER_ARB, numVertices * 3 *
sizeof(float), edgeCache, Gl.GL_DYNAMIC_DRAW_ARB);
    }
    else  // Existing lists
    {
        Gl.glBindBufferARB(Gl.GL_ARRAY_BUFFER_ARB, vertexVboName);
        Gl.glBufferSubDataARB(Gl.GL_ARRAY_BUFFER_ARB, 0, numVertices *
3 * sizeof(float), vertexCache);

        Gl.glBindBufferARB(Gl.GL_ARRAY_BUFFER_ARB, normalVboName);
        Gl.glBufferSubDataARB(Gl.GL_ARRAY_BUFFER_ARB, 0, numVertices *
3 * sizeof(float), normalCache);

        Gl.glBindBufferARB(Gl.GL_ARRAY_BUFFER_ARB, edgeVboName);
        Gl.glBufferSubDataARB(Gl.GL_ARRAY_BUFFER_ARB, 0, numVertices *
3 * sizeof(float), edgeCache);
    }
}
------

In my rendering method, I have the following code:
------
Gl.glEnableClientState(Gl.GL_VERTEX_ARRAY);

// Render faces
Gl.glEnable(Gl.GL_LIGHTING);
Gl.glEnableClientState(Gl.GL_NORMAL_ARRAY);
Gl.glBindBufferARB(Gl.GL_ARRAY_BUFFER_ARB, vertexVboName);
Gl.glVertexPointer(3, Gl.GL_FLOAT, 0, IntPtr.Zero);
Gl.glBindBufferARB(Gl.GL_ARRAY_BUFFER_ARB, normalVboName);
Gl.glNormalPointer(Gl.GL_FLOAT, 0, IntPtr.Zero);
Gl.glDrawArrays(Gl.GL_TRIANGLES, 0, mesh.Faces.Count * 3 * 3);
Gl.glDisableClientState(Gl.GL_NORMAL_ARRAY);
Gl.glDisable(Gl.GL_LIGHTING);

// Render edges
Gl.glColor4f(0.2f, 0.2f, 0.2f, 0.2f);
Gl.glEnable(Gl.GL_BLEND);
Gl.glBindBufferARB(Gl.GL_ARRAY_BUFFER_ARB, edgeVboName);
Gl.glVertexPointer(3, Gl.GL_FLOAT, 0, IntPtr.Zero);
Gl.glDrawArrays(Gl.GL_LINES, 0, mesh.Edges.Count * 2);
Gl.glDisable(Gl.GL_BLEND);

Gl.glDisableClientState(Gl.GL_VERTEX_ARRAY);
------
