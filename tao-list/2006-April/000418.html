Hello,
I searched in the archive with no luck, so here is my problem:
I'm trying to use Tao to triangulate complex polygons, but I'm not sure 
if I am using it correctly. The result seems incomplete.
Consider the simple program below. It should call callback function f 
setting param v with vertices coordinates, but the length of v is always 
1. The output is:
0
1
1
0
0

What am I doing wrong?

Thank you very much,
Stenio Brunetta

class Program {
    static void Main(string[] args) {
        double[] v0 = { 0, 0, 0 }; // A simple square
        double[] v1 = { 1, 0, 0 };
        double[] v2 = { 1, 1, 0 };
        double[] v3 = { 0, 1, 0 };

        Glu.GLUtesselator tess = Glu.gluNewTess();
        Glu.gluTessCallback(tess, Glu.GLU_TESS_VERTEX, new 
Glu.TessVertexCallback1(f));

        Glu.gluTessBeginPolygon(tess, IntPtr.Zero);
        Glu.gluTessBeginContour(tess);
        Glu.gluTessVertex(tess, v0, v0);
        Glu.gluTessVertex(tess, v1, v1);
        Glu.gluTessVertex(tess, v2, v2);
        Glu.gluTessVertex(tess, v3, v3);
        Glu.gluTessVertex(tess, v0, v0);

        Glu.gluTessEndContour(tess);
        Glu.gluTessEndPolygon(tess);

        Console.ReadLine();
    }

    private static void f(double[] v) {
       foreach(double d in v)
            Console.Write(d + " ");
        Console.WriteLine("");
    }
}
