Why not just package a small wrapping mcs.exe to stick in \Windows
with the install?

#include <string.h>
#include <windows.h>

int main(void) {
  char *cmdline=GetCommandLine();
  char *args=strchr(cmdline, ' ');

  char mcs[MAX_PATH]="C:\\mono\\mono.exe C:\\mono\\mcs.exe";
  if(args!=NULL) strcat(mcs, args);

  return system(mcs);
}
