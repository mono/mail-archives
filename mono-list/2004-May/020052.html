No it doesn't. Mono is just compiled with a broken C library on Windows.

#include <stdio.h>
#include <windows.h>

int main(int argc, char* argv[])
{
	printf("Raw: %s\n", GetCommandLine());
	for(int i =3D 0; i < argc; i++)
	{
		printf("%d: %s\n", i, argv[i]);
	}
	return 0;
}
