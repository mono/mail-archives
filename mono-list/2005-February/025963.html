MonoUnixSocket "/tmp/VirtualHosts/$appliacion/mono_socket/monosock"

before:
<apps>
<web-application>

         <vhost>a.host.name.com</vhost>
</web-application>
  ...
</apps>

and now:
<apps>
        <web-application>
                <name>$application</name>

	...
</apps>

Before:
