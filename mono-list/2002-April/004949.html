using System;
using System.Runtime.InteropServices;

namespace ConsoleApplication1
{
	/// <summary>
	/// Summary description for Class1.
	/// </summary>
	class Class1
	{
		// PostgreSQL client library:
		// under cygwin, it is pq.dll while under linux, it is libpq.so

		[DllImport("pq")]
		public static extern IntPtr PQconnectdb(String conninfo);
		// PGconn *PQconnectdb(const char *conninfo)

		[DllImport("pq")]
		public static extern void PQfinish(IntPtr conn);
		// void PQfinish(PGconn *conn)
	=09
		[DllImport("pq")]
		public static extern IntPtr PQexec(IntPtr conn,
			String query);
		// PGresult *PQexec(PGconn *conn,	const char *query);

		/// <summary>
		/// The main entry point for the application.
		/// </summary>
		[STAThread]
		static void Main(string[] args)
		{
			IntPtr dbconn;
			String sConnInfo;
			String sQuery;

			sConnInfo =3D "host=3Dlocalhost dbname=3Dtest user=3Ddanmorg =
password=3Dviewsonic";
			sQuery =3D "insert into sometable " +
				     "(tid,tdesc) " +
				     "values('pq','just pq') ";

			dbconn =3D PQconnectdb(sConnInfo);

			if(dbconn =3D=3D IntPtr.Zero)
				Console.WriteLine("dbconn is null");
			else
			{
				Console.WriteLine("assuming successful connection");
				PQexec(dbconn,sQuery);
				PQfinish(dbconn);
			}
		}
	}
}
