<code>

public partial class MainWindow
{
	private global::Gtk.UIManager UIManager;
        ...
	private global::Gtk.VBox vbox;
...
</code>

outside the gtk-gui folder:
<code>
public partial class MainWindow: Gtk.Window
{
...
