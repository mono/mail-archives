>   System.Diagnostics.Process proc = new System.Diagnostics.Process();

>   proc.StartInfo.Arguments = "machine -l mylogin \"./myscript\"";	
>   proc.StartInfo.UseShellExecute = false;

it may block when writing to StandardError.
