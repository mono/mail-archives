&gt;&gt; &gt; data table. &nbsp;If you try your example with<br>

&gt;&gt; data reader from the<br>
&gt;&gt; &gt; parameter. &nbsp;Then manually load a data table.<br>

&gt;&gt; &gt;<br>
&gt;&gt; &gt; // test.cs<br>
&gt;&gt; &gt; // gmcs test.cs /r:System.Data.dll<br>
&gt;&gt; /r:System.Data.OracleClient.dll<br>
&gt;&gt; &gt; using System;<br>
&gt;&gt; &gt; using System.Data;<br>
&gt;&gt; &gt; using System.Data.Common;<br>
&gt;&gt; &gt; using System.Data.OracleClient;<br>
&gt;&gt; &gt;<br>
&gt;&gt; &gt; public class test<br>
&gt;&gt; &gt; {<br>
&gt;&gt; &gt; &nbsp; &nbsp; public static void Main(string[] args)<br>
&gt;&gt; &gt; &nbsp; &nbsp; {<br>
&gt;&gt; &gt; &nbsp; &nbsp; &nbsp; &nbsp; Console.WriteLine(&quot;Start.&quot;);<br>
&gt;&gt; &gt; &nbsp; &nbsp; &nbsp; &nbsp; OracleConnection con = new<br>
&gt;&gt; OracleConnection(&quot;Data<br>
&gt;&gt; &gt; source=testdb;user<br>
&gt;&gt; id=someuser;password=somepass&quot;);<br>
&gt;&gt; &gt; &nbsp; &nbsp; &nbsp; &nbsp; con.Open();<br>
&gt;&gt; &gt; &nbsp; &nbsp; &nbsp; &nbsp; OracleCommand cmdSp = con.CreateCommand();<br>
&gt;&gt; &gt; &nbsp; &nbsp; &nbsp; &nbsp; cmdSp.CommandText = &quot;BEGIN\n<br>
&gt;&gt; SP_TEST_REF_CURSOR(:crsdata);<br>
&gt;&gt; &gt; \nEND;\n&quot;;<br>
&gt;&gt; &gt; &nbsp; &nbsp; &nbsp; &nbsp; DataSet ds = new DataSet();<br>
&gt;&gt; &gt; &nbsp; &nbsp; &nbsp; &nbsp; OracleParameter oprCursor = new<br>
&gt;&gt; OracleParameter(&quot;crsdata&quot;,<br>
&gt;&gt; &gt; OracleType.Cursor);<br>
&gt;&gt; &gt; &nbsp; &nbsp; &nbsp; &nbsp; oprCursor.Direction =<br>
&gt;&gt; ParameterDirection.Output;<br>
&gt;&gt; &gt; &nbsp; &nbsp; &nbsp; &nbsp; cmdSp.Parameters.Add(oprCursor);<br>
&gt;&gt; &gt;<br>

&gt;&gt; &gt;<br>
&gt;&gt; &gt; &nbsp; &nbsp;OracleDataReader reader =<br>
&gt;&gt; (OracleDataReader)oprCursor.Value;<br>

&gt;&gt; that holds a REF CURSOR&quot;);<br>
&gt;&gt; &gt;<br>
&gt;&gt; &gt; &nbsp; &nbsp;DataTable schema = reader.GetSchemaTable();<br>
&gt;&gt; &gt; &nbsp; &nbsp; &nbsp; &nbsp; DataTable newTable = new DataTable();<br>
&gt;&gt; &gt; &nbsp; &nbsp; &nbsp; &nbsp; if (reader.FieldCount &gt; 0)<br>
&gt;&gt; &gt; &nbsp; &nbsp; &nbsp; &nbsp; {<br>
&gt;&gt; &gt; &nbsp; &nbsp; &nbsp; &nbsp; &nbsp; &nbsp; foreach (DataRow row in schema.Rows)<br>
&gt;&gt; &gt; &nbsp; &nbsp; &nbsp; &nbsp; &nbsp; &nbsp; {<br>
&gt;&gt; &gt; &nbsp; &nbsp; &nbsp; &nbsp; &nbsp; &nbsp; &nbsp; &nbsp; string columnName =<br>

&gt;&gt; (Type)row[&quot;DataType&quot;];<br>
&gt;&gt; &gt; &nbsp; &nbsp; &nbsp; &nbsp; &nbsp; &nbsp; &nbsp; &nbsp; newTable.Columns.Add(columnName,<br>
&gt;&gt; columnType);<br>
&gt;&gt; &gt; &nbsp; &nbsp; &nbsp; &nbsp; &nbsp; &nbsp; }<br>
&gt;&gt; &gt; &nbsp; &nbsp; &nbsp; &nbsp; &nbsp; &nbsp; while (reader.Read())<br>
&gt;&gt; &gt; &nbsp; &nbsp; &nbsp; &nbsp; &nbsp; &nbsp; {<br>
&gt;&gt; &gt; &nbsp; &nbsp; &nbsp; &nbsp; &nbsp; &nbsp; &nbsp; &nbsp; object[] values = new<br>
&gt;&gt; object[newTable.Columns.Count];<br>
&gt;&gt; &gt; &nbsp; &nbsp; &nbsp; &nbsp; &nbsp; &nbsp; &nbsp; &nbsp; reader.GetValues(values);<br>
&gt;&gt; &gt; &nbsp; &nbsp; &nbsp; &nbsp; &nbsp; &nbsp; &nbsp; &nbsp; newTable.Rows.Add(values);<br>
&gt;&gt; &gt; &nbsp; &nbsp; &nbsp; &nbsp; &nbsp; &nbsp; }<br>
&gt;&gt; &gt; &nbsp; &nbsp; &nbsp; &nbsp; }<br>
&gt;&gt; &gt;<br>
&gt;&gt; &gt; &nbsp; &nbsp; &nbsp; &nbsp; reader.Close();<br>
&gt;&gt; &gt;<br>
&gt;&gt; &gt; &nbsp; &nbsp;Console.WriteLine(&quot;Add DataTable to<br>
&gt;&gt; DataSet&quot;);<br>
&gt;&gt; &gt; &nbsp; &nbsp;ds.Tables.Add(newTable);<br>
&gt;&gt; &gt;<br>
&gt;&gt; &gt; &nbsp; &nbsp;Console.WriteLine(&quot;Show Result&quot;);<br>
&gt;&gt; &gt; &nbsp; &nbsp; &nbsp; &nbsp; Console.WriteLine(&quot;Columns in<br>
&gt;&gt; DataTable&quot;);<br>
&gt;&gt; &gt; &nbsp; &nbsp; &nbsp; &nbsp; foreach (DataColumn column in<br>
&gt;&gt; ds.Tables[0].Columns)<br>
&gt;&gt; &gt; &nbsp; &nbsp; &nbsp; &nbsp; {<br>
&gt;&gt; &gt; &nbsp; &nbsp; &nbsp; &nbsp; &nbsp; &nbsp; Console.WriteLine(&quot; &nbsp;ColumnName:<br>
&gt;&gt; &quot; + column.ColumnName);<br>

&gt;&gt; &quot; +<br>
&gt;&gt; &gt; column.DataType.ToString());<br>
&gt;&gt; &gt; &nbsp; &nbsp; &nbsp; &nbsp; }<br>
&gt;&gt; &gt; &nbsp; &nbsp; &nbsp; &nbsp; Console.WriteLine(&quot;Rows in<br>
&gt;&gt; DataTable&quot;);<br>
&gt;&gt; &gt; &nbsp; &nbsp; &nbsp; &nbsp; foreach (DataRow row in ds.Tables[0].Rows)<br>
&gt;&gt; &gt; &nbsp; &nbsp; &nbsp; &nbsp; {<br>
&gt;&gt; &gt; &nbsp; &nbsp; &nbsp; &nbsp; &nbsp; &nbsp; for (int c = 0; c &lt;<br>
&gt;&gt; ds.Tables[0].Columns.Count; c++)<br>
&gt;&gt; &gt; &nbsp; &nbsp; &nbsp; &nbsp; &nbsp; &nbsp; {<br>
&gt;&gt; &gt; &nbsp; &nbsp; &nbsp; &nbsp; &nbsp; &nbsp; &nbsp; &nbsp; Console.WriteLine(&quot;Column &quot;<br>
&gt;&gt; + c.ToString());<br>
&gt;&gt; &gt; &nbsp; &nbsp; &nbsp; &nbsp; &nbsp; &nbsp; &nbsp; &nbsp; if (row[c].Equals(DBNull.Value))<br>
&gt;&gt; &gt; &nbsp; &nbsp; &nbsp; &nbsp; &nbsp; &nbsp; &nbsp; &nbsp; &nbsp; &nbsp; Console.WriteLine(&quot; &nbsp; &nbsp; Value<br>
&gt;&gt; is DBNull.Value&quot;);<br>
&gt;&gt; &gt; &nbsp; &nbsp; &nbsp; &nbsp; &nbsp; &nbsp; &nbsp; &nbsp; else<br>
&gt;&gt; &gt; &nbsp; &nbsp; &nbsp; &nbsp; &nbsp; &nbsp; &nbsp; &nbsp; {<br>
&gt;&gt; &gt; &nbsp; &nbsp; &nbsp; &nbsp; &nbsp; &nbsp; &nbsp; &nbsp; &nbsp; &nbsp; Console.WriteLine(&quot;<br>
&gt;&gt; Value: &quot; + row[c].ToString());<br>
&gt;&gt; &gt; &nbsp; &nbsp; &nbsp; &nbsp; &nbsp; &nbsp; &nbsp; &nbsp; &nbsp; &nbsp; Console.WriteLine(&quot; &nbsp; &nbsp; Value<br>
&gt;&gt; Type: &quot; +<br>
&gt;&gt; &gt; row[c].GetType().ToString());<br>
&gt;&gt; &gt; &nbsp; &nbsp; &nbsp; &nbsp; &nbsp; &nbsp; &nbsp; &nbsp; }<br>
&gt;&gt; &gt; &nbsp; &nbsp; &nbsp; &nbsp; &nbsp; &nbsp; }<br>
