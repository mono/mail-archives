Try boot.local<br>
/etc/init.d/boot.local<br>
<br>
Alternatively, here's a little template script so you can maybe control
this via YaST.. You need to of course fill in the stop and restart, but
I'll leave that as a challenge :)<br>
<br>
#!/bin/sh<br>
case "$1" in<br>
start)<br>
&nbsp;&nbsp;&nbsp; echo "Starting xsp..."<br>
&nbsp;&nbsp;&nbsp; nohup /path/to/mono /path/to/xsp.exe [yourarguments] --nonstop &amp;<br>
&nbsp;&nbsp;&nbsp; ;;<br>
stop)<br>
&nbsp;&nbsp;&nbsp; echo "Stopping xsp..."<br>
&nbsp;&nbsp;&nbsp; <br>
&nbsp;&nbsp;&nbsp; ;;<br>
restart)<br>
&nbsp;&nbsp;&nbsp; echo "Restarting xsp..."<br>
&nbsp;&nbsp;&nbsp; <br>
&nbsp;&nbsp;&nbsp; ;;<br>
*)<br>
&nbsp;&nbsp;&nbsp; echo "Usage: 'basename $0' {start|stop|restart}" &gt;&amp;2<br>
&nbsp;&nbsp;&nbsp; exit 64<br>
&nbsp;&nbsp;&nbsp; ;;<br>
<br>
<br>
esac<br>
exit 0<br>
