take a look at to <a class="moz-txt-link-freetext" href="http://www.mindspring.com/~matt.raffel/code/NET/">http://www.mindspring.com/~matt.raffel/code/NET/</a>&nbsp; <br>
<br>
There are 3 classes that might help you<br>
ApplicationCommandLine.cs<br>
CommandlineArgument.cs<br>
CommandlineException.cs<br>
<br>
To use them do something like this:<br>
<br>
<br>
First build an array of acceptable commandline arguements, such as ...<br>
<br>
<small><font face="Courier New, Courier, monospace"><br>
ApplicationCommandLine&nbsp;&nbsp; _cmdProcessor = new ApplicationCommandLine();<br>
<br>
// create the help argument<br>
_cmdProcessor.AddArg(new CommandLineArgument("?", "prints help"));<br>
<br>
// create the toggled argument of erase only or overwrite file options<br>
SwitchableCommandLineArgument eraseOnly = new
SwitchableCommandLineArgument("e", "only erases the file, content IS
NOT overwritten first");<br>
SwitchableCommandLineArgument overwriteFile = new
SwitchableCommandLineArgument("w", "overwrites the file prior to
erasing [default]");<br>
eraseOnly.SwitchArg = overwriteFile;<br>
overwriteFile.SwitchArg = eraseOnly;<br>
overwriteFile.Selected = true;<br>
_cmdProcessor.AddArg(eraseOnly);<br>
_cmdProcessor.AddArg(overwriteFile);<br>
<br>
// create an options<br>
_cmdProcessor.AddArg(new CommandLineArgument("z", "overwrites with
zeroes"));<br>
<br>
// create an option that takes data<br>
_cmdProcessor.AddArg(new DataCommandLineArgument("n", "number of times
the file should be overwritten"));<br>
<br>
<br>
<br>
<br>
then you need to process the command line by calling<br>
<br>
// assuming args is the string[] from main<br>
_cmdProcessor.ParseCmdLineToArgs(args);<br>
<br>
then you need to loop through to find the ones that have been set.&nbsp; I
do something like this<br>
<br>
foreach(CommandLineArgument argument in _cmdProcessor)<br>
{<br>
if (true == argument.Selected)<br>
{<br>
&nbsp;&nbsp;&nbsp; char ch = argument.Argument.ToString().ToCharArray()[0];<br>
&nbsp;&nbsp;&nbsp; switch (ch)<br>
&nbsp;&nbsp;&nbsp; {<br>
&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp; case 'z':<br>
&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp; // do something<br>
&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp; break;<br>
&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp; case '?':<br>
&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp; PrintHelp();<br>
&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp; break;<br>
&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp; default:<br>
&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp; throw new Exception(String.Format("found an weird arugment
{0}", ch));<br>
&nbsp;&nbsp;&nbsp; }<br>
}<br>
<br>
</font></small><br>
<br>
There maybe better ways but this works for me.&nbsp; I use it in my command
line tools all the time.&nbsp; Hope it helps...<br>
