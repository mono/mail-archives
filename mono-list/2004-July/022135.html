public sealed class HttpRequest {
.........
    private HttpWorkerRequest _WorkerRequest;
.........
    public bool IsSecureConnection {
            get {
                if (null != _WorkerRequest) {
                return _WorkerRequest.IsSecure();
            }
            return false;
        }
    }
