Try this:

private static int NumberOfSetBits (uint i)
{
	i = (i & 0x55555555) + ((i & 0xaaaaaaaa) >>  1);
	i = (i & 0x33333333) + ((i & 0xcccccccc) >>  2);
	i = (i & 0x0f0f0f0f) + ((i & 0xf0f0f0f0) >>  4);
	i = (i & 0x00ff00ff) + ((i & 0xff00ff00) >>  8);
	i = (i & 0x0000ffff) + ((i & 0xffff0000) >> 16);	return
(int) i;
}
