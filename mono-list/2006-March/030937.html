Try this (replace path/to/link/file with a valid path):

using System.Text;
public class main_t {
  public static void Main() {
    for (int i=0; i<10000; i++) {
      StringBuilder buf = new StringBuilder();
      Mono.Unix.Native.Syscall.readlink("path/to/link/file", buf);
    }
  }
}
