{
  if (!IsRealized)
  {
    return false;
  }

  Cairo.Context cr = Gdk.CairoHelper.Create(GdkWindow);
  bool retval = true;

  foreach (Gdk.Rectangle rect in evnt.Region.GetRectangles())

    cr.Clip();
