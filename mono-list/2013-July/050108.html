- (void)setDate:(NSDate *)value
{
    MonoObject *monoObject = [value monoDateTime];
    // [self setProperty:"Date" valueObject:monoObject]; // BAD
    [self setProperty:"Date" valueObject:(MonoObject *)mono_object_unbox(monoObject)];
}
