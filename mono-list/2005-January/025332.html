Label labelPopup = new Label("Click here to show a popup menu");

eventBox.ButtonPressEvent += new ButtonPressEventHandler(OnLabelPopupClick);

eventBox.Add(labelPopup);



popupMenu = new Menu();

MenuItem menuPopup1 = new MenuItem("Popup item 1");
