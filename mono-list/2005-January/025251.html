Both the dag wieers and nrpms.net yum/apt repositories have everything
you need to build the stable version of mono.

If you haven't added them to your config files, here are the yum.conf entries:

[dag]
name=Dag APT Repository
baseurl=http://dag.freshrpms.net/fedora/$releasever/en/$basearch/dag/
  http://dag.atrpms.net/fedora/$releasever/en/$basearch/dag/
  http://ftp.heanet.ie/pub/freshrpms/pub/dag/fedora/$releasever/en/$basearch/dag/
gpgcheck=1

[nr-production]
name=Fedora Core $releasever - $basearch - nr/production
baseurl=http://yum.nrpms.net/fedora-$releasever-$basearch/production

[nr-testing]
name=Fedora Core $releasever - $basearch - nr/testing
baseurl=http://yum.nrpms.net/fedora-$releasever-$basearch/testing
