/*

    OK

    mcs: tmp.cs(28) error CS0030: Cannot convert type 'int[,][]' to 'int[][,]'

using System;
class ArrayBug {
    public static int Main(String[] args) {
        int[][,] mb = new int[3][,];
        mb = (int[][,])mb;
        return 0;
    }
}
