The following test, for example, is compiled fine by mcs from cvs.

using System.Data;
class T {
        static IDataParameter get_param () {
                return null;
        }
        static void Main () {
                IDataParameter p = get_param ();
                if (p != null)
                        System.Console.WriteLine (p.ParameterName);
        }
}
