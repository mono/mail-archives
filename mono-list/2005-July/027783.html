Have a look in the dot Directory .config, go to ~/.congig/MonoDevelop
and you will find several xml files, the one you are interested in is
DefaultEditingLayout.XML, this is the config file that controls the
layout of the IDE. I have all my panels docked on the left.

On my system it looks like this

<dock-layout>
  <layout name="__default__">
    <dock floating="no" floatx="0" floaty="0" height="-1" width="-1"
name="__dock_1">
      <paned position="209" locked="no" orientation="horizontal"
name="__dock_9">
        <notebook page="3" locked="no" orientation="vertical"
name="__dock_10">
          <item locked="no" orientation="vertical"
name="MonoDevelop.Gui.Pads.HelpTree" />
          <item locked="no" orientation="vertical"
name="MonoDevelop.Gui.Pads.FileScout" />
          <item locked="no" orientation="vertical"
name="MonoDevelop.Gui.Pads.ClassPad" />
          <item locked="no" orientation="vertical"
name="MonoDevelop.Gui.Pads.ProjectPad" />
          <item locked="no" orientation="vertical"
name="MonoQuery.Pads.MonoQueryView" />
        </notebook>
        <paned position="530" locked="no" orientation="vertical"
name="__dock_11">
          <item locked="yes" orientation="vertical" name="Documents" />
          <item locked="no" orientation="vertical"
name="MonoDevelop.Gui.Pads.OpenTaskView" />
        </paned>
      </paned>
    </dock>
  </layout>
  <layout name="Edit.Default">
    <dock floating="no" floatx="0" floaty="0" height="-1" width="-1"
name="__dock_1">
      <paned position="209" locked="no" orientation="horizontal"
name="__dock_9">
        <notebook page="3" locked="no" orientation="vertical"
name="__dock_10">
          <item locked="no" orientation="vertical"

          <item locked="no" orientation="vertical"

          <item locked="no" orientation="vertical"

        </notebook>
        <paned position="530" locked="no" orientation="vertical"
name="__dock_11">
          <item locked="yes" orientation="vertical" name="Documents" />
          <item locked="no" orientation="vertical"

</dock-layout>
