like :
   
#cairo
deb http://cairographics.org/packages/debian/ ./
deb-src http://cairographics.org/packages/debian/ ./
#mono
deb http://www.meebey.net/debian/monodevelop ./
deb http://www.meebey.net/debian/gtksourceview-sharp ./
deb http://www.meebey.net/debian/gecko-sharp ./

#deb http://linux.cict.fr/ unstable main non-free contrib
#deb-src http://linux.cict.fr/ unstable main non-free contrib

#deb http://ftp.tiscali.de/pub/debian/debian/ stable main  contrib
#deb-src http://ftp.tiscali.de/pub/debian/debian/ stable main  contrib

#deb http://ftp.tiscali.de/pub/debian/debian/ unstable main  contrib
#deb-src http://ftp.tiscali.de/pub/debian/debian/ unstable main  contrib

# deb http://security.debian.org/ stable/updates main contrib non-free

deb http://ftp-stud.fht-esslingen.de/pub/Mirrors/ftp.debian.org/debian/ unstable main  contrib
deb-src http://ftp-stud.fht-esslingen.de/pub/Mirrors/ftp.debian.org/debian/ unstable main contrib

and type 
     apt-get update
     apt-get dist-upgrade
