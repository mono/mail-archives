		public static int Main (string[] args) {
			string imageFilename = "SqlSharpForGtkSharpSplash.png";
			Application.Init ();
			SplashScreen splash = new SplashScreen (
				SqlSharpGtk.ApplicationName,
				imageFilename);

			sqlSharp.Show ();
			splash.Destroy ();
			Application.Run ();

		}

a good approach might be:

* start the program an immediately create the splash screen, and install
an idle handler
* do all initialization and start GTK main loop
* when the idle callback is called, create the main window and destroy
the spash screen.

If that's too quick, and the splash screen is shown and destroyed too
quickly, you can use a timer instead of the idle handler.

using System;
using Gtk;
using GtkSharp;

namespace Mono.GtkSharp.Goodies=20
{
	public class SplashScreen=20
	{
		Window window;
		VBox vbox;
		Gtk.Image image;
	=09
		public SplashScreen (string applicationName, string imageFilename)=20
		{
			window =3D new Window (applicationName);=0A=
			window.DeleteEvent +=3D new DeleteEventHandler (Window_Delete);=0A=
			window.DefaultSize =3D new System.Drawing.Size (200, 200);=0A=
=0A=
			vbox =3D new VBox (false, 0);=0A=
			window.Add (vbox);=0A=
=0A=
			try {=0A=
				image =3D new Gtk.Image (imageFilename);=0A=
			}=0A=
			catch (Exception e) {=0A=
				Console.WriteLine("Error: Image file not found.");=0A=
			}=0A=
			vbox.PackStart (image, false, false, 0);=0A=
			window.ShowAll ();
			while (Application.EventsPending ())=20
				Application.RunIteration ();
		}

		public void Destroy () {
			if (window !=3D null) {
				window.Destroy ();
				window =3D null;
			}
		}

		void Window_Delete (object o, DeleteEventArgs args) {=0A=
			if (window !=3D null) {=0A=
				window.Destroy ();=0A=
				window =3D null;=0A=
			}=0A=
		}=0A=

	}
}
