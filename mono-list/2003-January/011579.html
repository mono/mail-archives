    <?xml version="1.0" encoding="iso-8859-1"?>
    
    <project name="hw" default="build">
    	<property name="debug" value="true"/>
    	
    	<target name="build">
    		<mcs target="exe" output="hw.exe" debug="${debug}">
    			<sources basedir="src">

    			</sources>
    			<references>
    				<includes name="gtk-sharp.dll"/>

    		</mcs>
    	</target>
    	
    	<target name="clean">

    	</target>
