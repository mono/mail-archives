Here you should simply switch on the simple type in element class:

	switch (klass->element_class->byval_arg.type) {

	case MONO_TYPE_I2:
	case MONO_TYPE_U2:
		SWAP (16);
		break;
	case MONO_TYPE_I4:
	case MONO_TYPE_U4:
