Whoops, I guess that was an extension method I wrote:

public static void DrawPixbuf (this Context g, Gdk.Pixbuf pixbuf, Point 
dest)
{
	g.Save ();

	Gdk.CairoHelper.SetSourcePixbuf (g, pixbuf, dest.X, dest.Y);
	g.Paint ();
	g.Restore ();
}
