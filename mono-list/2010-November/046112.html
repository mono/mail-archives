Draw your clipping rectangle path, call Clip, then set up and stroke 
your line:

cc = Gdk.CairoHelper.Create (drawingarea1.GdkWindow);
cc.Rectangle (50,50,80,80);
cc.Clip ();

cc.Color = new Cairo.Color (0,0,0);
cc.MoveTo (10,10);
cc.LineTo (100, 100);
cc.Stroke();

Jonathan
