> <dllmap dll="glibsharpglue-2"
> target="/Library/Frameworks/Mono.framework/Versions/Current/lib/libglibsharpglue-2.so"

> target="/Library/Frameworks/Mono.framework/Versions/Current/lib/libgobject-2.0.0.dylib"
