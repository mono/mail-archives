Btw. if there are any more Emacs/Gnus users on this list:

====
(defun gnus-simplify-subject-martin (subject)
  ""
  (save-excursion
    (gnus-set-work-buffer)
    (let ((case-fold-search t))
      (insert subject)
      (gnus-simplify-buffer-fuzzy-step "\\[\\(.*-list\\|.*-patches\\)\\]")
      (gnus-simplify-buffer-fuzzy-step "\t" " ")
      (gnus-simplify-buffer-fuzzy-step "  +" " ")
      (gnus-simplify-buffer-fuzzy-step " $")
      (gnus-simplify-buffer-fuzzy-step "^ +")
      (buffer-string))))

(setq gnus-select-group-hook
      (list
       (lambda ()
	 (mapcar (lambda (header)
		   (mail-header-set-subject
		    header
		    (gnus-simplify-subject-martin
		     (mail-header-subject header))))
		 gnus-newsgroup-headers))))
