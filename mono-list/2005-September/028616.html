dist)

    ;;
all)

    ;;

ARCH=%{_target_platform}

case "$RPM_COMMAND" in
prepare|all)
    
	CFLAGS="-O2 " ./configure --prefix=/usr --libdir=/usr/lib64 --sysconfdir=/etc --mandir=/usr/share/man --infodir=/usr/info --localstatedir=/var  --with-preview=yes --with-signaltstack=no
      
    ;;
esac

clean|all)

    fi
    ;;
esac

build|all)

    ;;
esac
