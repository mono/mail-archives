Try the version from svn:

svn co svn://svn.myrealbox.com/source/trunk/cocoa-sharp
cd cocoa-sharp
export ACLOCAL_FLAGS="-I /Library/Frameworks/Mono.framework/Versions/ 
Current/share/aclocal"
./autogen.sh --prefix=/Library/Frameworks/Mono.framework/Versions/ 
Current
make
sudo make install
