The following, perhaps not the most elegant, worked
for me.

Yair

[Export("showDialog:")]
public void showAbout(Cocoa.Object sender) 
{
	Cocoa.String[] vKeys    = new Cocoa.String[3];
	Cocoa.Object[] vObjects = new Cocoa.Object[3];
	vKeys[0] = new Cocoa.String("ApplicationVersion");
vObjects[0] = new Cocoa.String("Version 4");
	vKeys[1] = new Cocoa.String("Version"           );
vObjects[1] = new Cocoa.String("4.00"     );
	vKeys[2] = new Cocoa.String("ApplicationIcon"   );
vObjects[2] = new Cocoa.Image ("KO.icns"  );
	ObjCMessaging.objc_msgSend(
		Application.SharedApplication.NativeObject,
"orderFrontStandardAboutPanelWithOptions:", 
		typeof(void), typeof(IntPtr), new
Cocoa.cDictionary(vKeys, vObjects).NativeObject);
}
