Since your getting rid of ugliness, rather than passing these  
IntPtr's around everywhere why not make a Interface ID that defines a  
wrapped Native Object such as this:

namespace Cocoa
{
	public interface ID
	{
		IntPtr NativeObject{
			get;
		}
	}
}

And have both ObjCClass & Object support it:

namespace Cocoa {
	public class ObjCClass:ID
	{
		public IntPtr NativeObject
		{
			get {
				return ToIntPtr();
			}
		}

		[...]
	}
}
namespace Cocoa {
	public class Object : IDisposable,ID {
  		[...]
	}
}


Then you can have an objc_msgSend that takes ID as an argument for  
the receiver, and the actual return type as the return type

namespace Cocoa {
	public class ObjCMessaging {

		public static object objc_msgSend (ID receiver, string selector,  
Type rettype) {
			if(typeof(ID).IsAssignableFrom(rettype)
			   || typeof(IntPtr) ==rettype){
				return Object.FromIntPtr((IntPtr)objc_msgSend  
(receiver.NativeObject,  selector, typeof(IntPtr)));
			}else{
				return objc_msgSend (receiver.NativeObject,  selector, rettype);
			}
		}

		public static object objc_msgSend (ID receiver, string selector,  
Type rettype, params object[] args) {
			if(typeof(ID).IsAssignableFrom(rettype)
			   || typeof(IntPtr) ==rettype){
				return Object.FromIntPtr((IntPtr)objc_msgSend  
(receiver.NativeObject, selector, typeof(IntPtr), args));
			}else{
				return objc_msgSend (receiver.NativeObject, selector, rettype,  
args);
			}
		}

		[...]
	}

}

And when dealing with args inside the objc_msgSend that takes an IntPtr
check when typeof(ID).IsAssignableFrom(args[i]) and you can replace  
it with typeof(IntPtr)
and when args[i+1] is ID you can call the NativeObject  property to  
set the realArg,



This makes implementing Native Object wrappers methods look much cleaner

namespace Cocoa {
	public class OpenPanel : SavePanel {
		[...]

		public bool CanChooseDirectories {
			get {
				return (bool) ObjCMessaging.objc_msgSend (this,  
"canChooseDirectories", typeof (bool));
			}
			set {
				ObjCMessaging.objc_msgSend (this, "setCanChooseDirectories:",  
typeof (void),typeof (bool), value);
			}
		}

		[...]
}
namespace Cocoa {
	public class MutableArray : Array {

		[...]

		public void Add (ID o) {
			ObjCMessaging.objc_msgSend (this, "addObject:", typeof (void),  
typeof(ID),o);
		}
	}
}


namespace Cocoa {
	public class OutlineView : Control {
		[...]

		public ID ItemAtRow (int row) {
			return (ID) ObjCMessaging.objc_msgSend (this, "itemAtRow:", typeof  
(ID), typeof (int), row);
		}

		[...]
	}
}

namespace Cocoa {
	public class Color : Cocoa.Object {
		[...]
		
		public static Color Black {
			get {
				return (Color) ObjCMessaging.objc_msgSend (ObjCClass.FromType  
(typeof (Color)), "blackColor", typeof (Color));
			}
		}	
		[...]
	}
}
