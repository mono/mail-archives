i edit the makefile for myself. Now it will be compiled sucessfully.  
Here we go:

APP=CurrencyConverter.app

all: test

test: CurrencyConverter.cs
	rm -Rf $(APP)
	mkdir -p $(APP)/Contents/Resources
	mcs CurrencyConverter.cs -out:$(APP)/Contents/Resources/Test.exe  
-r:/Library/Frameworks/Mono.framework/Versions/1.0/lib// 
Apple.Foundation.dll  
-r:/Library/Frameworks/Mono.framework/Versions/1.0/lib/Apple.AppKit.dll
	mkdir -p $(APP)/Contents/MacOS
	#cp ../*.icns $(APP)/Contents/Resources
	cp ./Info.plist $(APP)/Contents/Info.plist
	cp /Library/Frameworks/Mono.framework/Versions/1.0/lib/*.dll*  
$(APP)/Contents/Resources
	cp /Library/Frameworks/Mono.framework/Versions/1.0/lib/*Glue*  
$(APP)/Contents/Resources
	cp  
/Library/Frameworks/Mono.framework/Versions/1.0/bin/CocoaSharpLoader  
$(APP)/Contents/MacOS/CurrencyConverter
	cp -r CurrencyConverter.nib $(APP)/Contents/Resources

run: test
	./$(APP)/Contents/MacOS/CurrencyConverter
