Dolapo,

Here is the Makefile (and the config file from
Mono.framework/Versions/Current/etc/mono). Hope they are of some help.

Regards
Gareth

LIB=/Library/Frameworks/Mono.framework/Libraries/mono/cocoa-sharp
GLUE=/Library/Frameworks/Mono.framework/Libraries
BIN=/Library/Frameworks/Mono.framework/Commands
APP=CocoaDrawing.app

all: test

test: CocoaDrawing.cs
	rm -Rf $(APP)
	mkdir -p $(APP)/Contents/Resources/English.lproj
	mcs -g CocoaDrawing.cs -out:$(APP)/Contents/Resources/Test.exe
-pkg:cocoa-sharp
	mkdir -p $(APP)/Contents/MacOS
	cp ./mono.icns $(APP)/Contents/Resources
	cp ./Info.plist $(APP)/Contents/Info.plist
	cp $(GLUE)/*Glue* $(APP)/Contents/Resources
	cp $(BIN)/CocoaSharpLoader $(APP)/Contents/MacOS/CocoaDrawing
	cp -r ./English.lproj/MainMenu.nib
$(APP)/Contents/Resources/English.lproj

run: test
	./$(APP)/Contents/MacOS/CocoaDrawing

and the config file:

<configuration>
	<dllmap dll="cygwin1.dll" target="libc.dylib" />
	<dllmap dll="libc" target="libc.dylib" />
	<dllmap dll="libintl" target="libintl.dylib" />
	<dllmap dll="libxslt.dll" target="libxslt.dylib" />
	<dllmap dll="libmySQL.dll" target="libmysqlclient.dylib" />
	<dllmap dll="odbc32.dll" target="libodbc.dylib" />
	<dllmap dll="oci" target="clntsh" />
	<dllmap dll="db2cli" target="libdb2_36.dylib" />
	<dllmap dll="libX11.so" target="libX11.dylib" />
	<dllmap dll="libX11" target="libX11.dylib" />
	<dllmap dll="MonoPosixHelper" target="libMonoPosixHelper.dylib" />
	<dllmap dll="gdiplus.dll" target="libgdiplus.dylib" />
	<dllmap dll="CairoImp" target="libcairo.dylib" />
	<dllmap dll="Glue" target="libCocoaSharpGlue.dylib" />
	<dllmap dll="FoundationGlue" target="libFoundationGlue.dylib" />
</configuration>

Note the two lines for libX11, I had to add the second one last night after
a cvs update of mcs last night!!
