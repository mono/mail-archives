using System;
using Gtk;
using Glade;
using Gecko;

public class Prueba
{
	[Glade.Widget] private Gnome.App app1;
	[Glade.Widget] private ScrolledWindow swPresentation;
        private Gnome.Program program;

	public static void Main (string[] args)
	{
		new Prueba (args);
	}

	public Prueba (string[] args)
	{
		program = new Gnome.Program ("mCatalog", "1.0", Gnome.Modules.UI, args);
		Glade.XML gxml = new Glade.XML (null, "prueba.glade", "app1", null);
		gxml.Autoconnect (this);

		WebControl wc = new WebControl ();
		swPresentation.AddWithViewport (wc);
		app1.ShowAll ();
	}
}
