Hello,

OK, the Gnome# Hello World program has been revised once more
thanks to a suggestion from Martin Willemoes Hansen.

class Hello
{
    static void Main(string[] args)
    {
        Gnome.Program program =
        new Gnome.Program("Hello World", "1.0", Gnome.Modules.UI, args);

        Gnome.App app = new Gnome.App("Hello World", "Hello World");
        app.Show();
 
        program.Run();
    }
}


See ya
