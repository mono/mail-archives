Check Gtk.TreeView. Here follows a simple example


<snippet>

TreeView tv =3D new TreeView ();
TreeStore store =3D new TreeStore (typeof (string));
TreeIter iter, iter2;

int counter, counter2, counter3;
int max =3D 5;
	=09
// populate the treeview
counter =3D 0;
while (counter < max)
{
	iter =3D store.AppendValues ((string) counter);
=09
	counter2 =3D 0;
	while (counter2 < max)
	{
		iter2 =3D store.AppendValues (iter, (string) counter2);
		counter2++;

		counter3 =3D 0;
		while (counter3 < max)
		{
			store.AppendValues (iter2, (string) counter3);
			counter3++;
		}
	}

	counter++;
}

tv.Model =3D store;
tv.ExpandAll(); // this is optional wether you want to Expand or not the
TreeView automatically

</snippet>
