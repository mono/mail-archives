What I suggested works:

using System;
using Gtk;

class T : Window
{
     static void Main ()
     {
		  Application.Init ();
          T t = new T();
		  t.ShowAll();
          Application.Run ();
	 }

     T () : base ("HALLO")
     {
          SetDefaultSize (400, 400);
     }

     protected override bool OnDeleteEvent ( Gdk.Event e )
     {
          Console.WriteLine("HI");
		  return true;
     }
}
