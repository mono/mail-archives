You can do this in Gtk# for Windows 1.0.2

Gtk.Widget w =  gxml.GetWidget("combo1");
combo1 = w as Gtk.Combo;
string[] list = new string[] {"English", "Maths", "Science"};
combo1.PopdownStrings = list;
