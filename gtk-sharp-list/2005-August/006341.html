using System;
using Gdk;
using Gtk;
using Gnome;

namespace VEdit.UI.GtkUi {

  public class Utils {
	
    public static HBox CreatePicLabelWidget(String pic, String lab) {
      HBox hb = new HBox();

      hb.PackStart(new Gtk.Image(new Gdk.Pixbuf(null, pic)), false,
false, 0);
      hb.PackStart(new Label(lab), true, true, 5);
			
      return hb;
    }
  }
}
