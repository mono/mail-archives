        {
                int resp =3D this.Run ();
               =20
                this.Hide ();
                       =20
                if (resp !=3D (int) Gtk.ResponseType.Ok)
                        throw new Exception ("No File Selected.");

        }

        {
                FileDialog fd =3D new FileDialog (title,
