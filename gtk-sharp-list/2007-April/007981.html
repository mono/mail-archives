private bool ToDoFilter(Gtk.TreeModel _model, Gtk.TreeIter _iter)
{
  Task task;
  Console.WriteLine("Filtering task for ToDo list");
  task = (Task)_model.GetValue(_iter, 0);
  if (task == null) return false;
  if (task.IsToDo) return true;
  return false;
}
