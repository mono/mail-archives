void setColor()
	{

		colorDialog = new ColorSelectionDialog("select color");
		
		
		
		if(colorDialog.Run()==(int)ResponseType.Ok)
		{
			selected = colorDialog.ColorSelection;			
			
			textview1.ModifyText(StateType.Normal,selected.CurrentColor);		
			
		}	
		colorDialog.Destroy();
		
	}
