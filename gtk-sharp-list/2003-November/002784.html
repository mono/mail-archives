namespace NS {
using System;
public class Test : IDisposable
{
	public void Dispose ()
	{
		GC.SuppressFinalize (this);
		Console.WriteLine ("Disposed");
	}

	~Test ()
	{
		Console.WriteLine ("Finalized");
	}

	public static void Main ()
	{
		for (long i = 10000000000; i > 0; i--)
 			new Test ();	
	}	
}
}
