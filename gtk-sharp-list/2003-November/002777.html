using Gtk;
using GtkSharp;

class test

	{

		Application.Init();

		Window MainWindow = new Window(":: this is just a test ::");
		MainWindow.DefaultWidth = 256;
		MainWindow.DeleteEvent += new DeleteEventHandler(OnMainWindowDelete);

	}

	{
		Application.Quit ();

	{
		Application.Quit ();
	}
