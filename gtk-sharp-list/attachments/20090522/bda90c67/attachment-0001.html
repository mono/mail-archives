Here is what worked well for me, if a more verbose example would help
please feel free to ask. I didn't test this example but I'm pretty sure
it's correct.<br>
<br>
TreeView1.ButtonPressEvent += new
ButtonPressEventHandler(TreeView1_ButtonPress);&nbsp;&nbsp;&nbsp; <br>
<br>
[GLib.ConnectBefore]<br>
private void TreeView1_ButtonPress(object o, Gtk.ButtonPressEventArgs
args)<br>
{&nbsp;&nbsp;&nbsp; <br>
&nbsp;&nbsp;&nbsp; // check if the right button was pushed<br>
&nbsp;&nbsp;&nbsp; if((int)args.Event.Button == 3)<br>
&nbsp;&nbsp;&nbsp; {&nbsp;&nbsp;&nbsp; &nbsp;&nbsp;&nbsp; <br>
&nbsp;&nbsp;&nbsp; &nbsp;&nbsp;&nbsp; Gtk.Menu jBox = new Gtk.Menu();<br>
&nbsp;&nbsp;&nbsp; &nbsp;&nbsp;&nbsp; Gtk.MenuItem MenuItem1 = new MenuItem("new job");<br>
&nbsp;&nbsp;&nbsp; &nbsp;&nbsp;&nbsp; jBox.Add(MenuItem1);&nbsp;&nbsp;&nbsp; &nbsp;&nbsp;&nbsp; <br>
&nbsp;&nbsp;&nbsp; &nbsp;&nbsp;&nbsp; MenuItem1 = new MenuItem("edit job");<br>
&nbsp;&nbsp;&nbsp; &nbsp;&nbsp;&nbsp; jBox.Add(MenuItem1);<br>
&nbsp;&nbsp;&nbsp; &nbsp;&nbsp;&nbsp; MenuItem1 = new MenuItem("remove job");<br>
&nbsp;&nbsp;&nbsp; &nbsp;&nbsp;&nbsp; jBox.Add(MenuItem1);<br>
&nbsp;&nbsp;&nbsp; &nbsp;&nbsp;&nbsp; MenuItem1 = new MenuItem("remove finished jobs");<br>
&nbsp;&nbsp;&nbsp; &nbsp;&nbsp;&nbsp; jBox.Add(MenuItem1);&nbsp;&nbsp;&nbsp; &nbsp;&nbsp;&nbsp; &nbsp;&nbsp;&nbsp; <br>
&nbsp;&nbsp;&nbsp; &nbsp;&nbsp;&nbsp; jBox.ShowAll();<br>
&nbsp;&nbsp;&nbsp; &nbsp;&nbsp;&nbsp; jBox.Popup();<br>
&nbsp;&nbsp;&nbsp; }<br>
}<br>
