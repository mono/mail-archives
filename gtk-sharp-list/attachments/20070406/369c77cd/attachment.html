<br><br>int width = 0;<br>int height = 0;<br><br>// get the root window<br>Gdk.Window root = Gdk.Global.DefaultRootWindow;<br><br>// get its width and height<br>root.GetSize(out width, out height);<br><br>// create a new pixbuf from the root window
<br>Gdk.GdkPixbuf screenshot = Gdk.GdkPixbuf.CreateFromDrawable(root, root.Colormap, 0, 0, 0, 0, width, height);<br><br>Dave<br>
