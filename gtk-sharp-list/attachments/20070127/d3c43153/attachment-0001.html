No. The dialog creation part is almost the same. There is nothing about
filters or selecting multiple files. This example is in monodoc:<br>
<br>
public class MainWindow: Gtk.Window {<br>
     <br>
     protected virtual void OnBtnLoadFileClicked(object sender,
System.EventArgs e)<br>
     {<br>
          Gtk.FileChooserDialog fc=<br>
          new Gtk.FileChooserDialog("Choose the file to open",<br>
                                      this,<br>
                                      FileChooserAction.Open,<br>
                                      "Cancel",ResponseType.Cancel,<br>
                                      "Open",ResponseType.Accept);<br>
<br>
          if (fc.Run() == (int)ResponseType.Accept) <br>
          {<br>
               System.IO.FileStream
file=System.IO.File.OpenRead(fc.Filename);<br>
               file.Close();<br>
          }<br>
          //Don't forget to call Destroy() or the FileChooserDialog
window won't get closed.<br>
          fc.Destroy();<br>
     }
