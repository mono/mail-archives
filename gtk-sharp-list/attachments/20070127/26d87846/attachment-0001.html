<div>Here is how:<br>
// create the dialog<br>
// remember to choose correct FileChooserAction from these
self-explanatory values:<br>
// Open, Save, SelectFolder, CreateFolder<br>
FileChooserDialog Fcd = new FileChooserDialog ("Open file", null,
FileChooserAction.Open);<br>
<br>
// add buttons you wish to see in the dialog<br>
Fcd.AddButton(Stock.Cancel, ResponseType.Cancel);<br>
</div>
Fcd.AddButton(Stock.Open, ResponseType.Ok);<br>
<br>
// then create a filter for files. For example .gif:<br>
// filter is not necessary if you wish to see all files in the dialog<br>
Fcd.Filter = new FileFilter();<br>
Fcd.Filter.AddPattern("*.gif");<br>
<br>
// if you wish to select multiple files set the do the following:<br>
Fcd.SelectMultiple = true;<br>
// In this case read Filenames-property instead of Filename-property<br>
<br>
// run the dialog<br>
ResponseType RetVal = (ResponseType)Fcd.Run();<br>
<br>
// handle the dialog's exit value<br>
// Read the file name from Fcd.Filename<br>
if (RetVal == ResponseType.Ok) {<br>
// do something<br>
} else {<br>
// do something else<br>
}<br>
<br>
// destroy the dialog<br>
Fcd.Destroy();<br>
<br>
PS. There is a documentation covering gtk# at <a class="moz-txt-link-freetext" href="http://docs.gotmono.net/">http://docs.gotmono.net/</a>.<br>
