> Or in short form:
>
> UpdateStatusbar (Gtk.TextBuffer buffer, Gtk.Statusbar statusbar)
> {
>    statusbar.Pop (0);
>    Gtk.TextIter iter = buffer.GetIterAtMark (buffer.InsertMark);
>    string msg = string.Format ("Col {0} Ln {1}", iter.Line + 1,

>    statusbar.Push (0, msg);
