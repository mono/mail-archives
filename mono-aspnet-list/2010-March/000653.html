   B - fastcgi server

exec env -i \
PATH="/home/user/mono/bin:$PATH" \

MONO_SHARED_DIR="/home/user/tmp" \
/home/user/mono/bin/fastcgi-mono-server2 /socket=tcp /port=4793 \
/address=0.0.0.0 /applications=/:/home/user/www /root=/home/user/www \

modify .htaccess file.
