Sure just create your own method. Here I do a query, get the schema, iterate
through and create my dataset, then fill it.

public static DataSet RunQuery (string query) {
    GetCon (); //sets a static variable conn with the connection info

    try {
        //DUPLICATE THE DataAdapter.Fill Method since it's not available in
Mono Android.Net
        //create a dataset to hold the results
        DataSet ds = new DataSet();
        ds.Tables.Add();

        //do the query to get the data
        Mono.Data.Sqlite.SqliteCommand c = new SqliteCommand ();
        c.CommandText = query;
        c.Connection = conn;
        System.Data.Common.DbDataReader reader = c.ExecuteReader();

        //create the columns in the dataset based on the schema of the table
        DataTable dt = reader.GetSchemaTable();
        foreach(DataRow dr in dt.Rows) {
            ds.Tables[0].Columns.Add(dr["ColumnName"].ToString());
        }

        //now add the data
        while(reader.Read()) {
            //copy column data to a temp array
            string[] vals = new string[ds.Tables[0].Columns.Count];
            for(int i=0;i<vals.Length;i++) {
                vals[i] = reader[i].ToString();
            }

            //add new data to dataset
            ds.Tables[0].Rows.Add(vals);
        }

        //return the filled dataset
        return ds;
    } catch (Exception ex) {
        Console.WriteLine("RunQuery() error running query: " + ex.Message);			
    }

    return null;
}
