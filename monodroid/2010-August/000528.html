        public XmlDocument ToXmlDocument()
        {
            XmlDocument doc = new XmlDocument();
            XmlElement root = doc.CreateElement("RestArgs");
            doc.AppendChild(root);

            // walk...
            foreach (string name in this.Keys)
            {

                root.AppendChild(element);

            }

            // return...
            return doc;
        }
