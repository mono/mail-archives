Seems to be a problem with ItemClick in the tutorial (?). The following works fine;

  protected override void OnListItemClick(ListView l, View v, int position, long id)
        {
            System.Diagnostics.Debug.WriteLine("OnListItemClick called");
            Title = "OnItem Clicked";
            Toast.MakeText(Application, ((TextView)v).Text, Toast.LengthShort).Show();
            base.OnListItemClick(l, v, position, id);

        }
