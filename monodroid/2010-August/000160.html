Just got the link to MonoDroid today, and thought I would quickly just plug in a call to a REST-based URL I have been playing with lately.

However, it appears that on the 'GetResponse()' of the Request object, I get "An exception was thrown by the type initializer for System.Net.WebConnection"

Below is the full code, the URL that it hits is live so you should be able to run this just as it is.

The same code runs fine in a sample page on my .NET website , so I /hope/ it's not just me typing something wrong.

Ken

using System;
using Android.App;
using Android.Content;
using Android.Runtime;
using Android.Views;
using Android.Widget;
using Android.OS;
using System.Net;
using System.IO;
namespace MonoDroid_Hello
{
public class Activity1 : Activity
{
Button button;
public Activity1(IntPtr handle)
: base(handle)
{
}
protected override void OnCreate(Bundle bundle)
{
base.OnCreate(bundle);
// Create your application here
button = new Button(this);

button.Text = "Click Me!";
button.Click += delegate { GetPrayer(); };
SetContentView(button);
}
public void GetPrayer()
{
try
{
HttpWebRequest request = (HttpWebRequest)WebRequest.Create("http://www.entropyspecialists.com/services/PrayerCookieService.svc/prayer");
request.Method = "GET";
var response = (HttpWebResponse)request.GetResponse();
Stream receiveStream = response.GetResponseStream();
StreamReader reader = new StreamReader(receiveStream);
string prayerText = reader.ReadToEnd();
button.Text = prayerText;
}
catch (Exception ex)
{
button.Text = ex.Message + ":" + ex.StackTrace;
}
}
}
}
