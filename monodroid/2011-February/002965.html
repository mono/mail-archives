Sorry about not including that code. Wasn't sure which code would be of
interest here. Below I've provided my implementation of a listener. Can I
ask what you mean by deriving from Java.Lang.Object? Are you saying that we
should always inherit from that on all of our classes, or just listeners?
Either way, why? I'm afraid I don't really understand the adapter/activation
system that is taking place here. Maybe if I did, I would be able to avoid
these problems in the future.

public class PageAnimationListener : Animation.IAnimationListener
    {
        private PageAnimationBase m_animation;
        int m_nCount = 0;

        public PageAnimationListener()
        { }

        public PageAnimationListener(IntPtr _ptr)
        { }

        public PageAnimationListener(PageAnimationBase _animation)
        {
            m_animation = _animation;
        }

        #region IAnimationListener Members

        public void OnAnimationEnd(Animation animation)
        {
            m_nCount++;

            if (m_nCount >= 2)
            {
                m_animation.FinishAnimating();
            }
        }

        public void OnAnimationRepeat(Animation animation)
        {
        }

        public void OnAnimationStart(Animation animation)
        {
        }

        #endregion

        #region IJavaObject Members

        public IntPtr Handle
        {
            get;
            set;
        }

        #endregion
    }
