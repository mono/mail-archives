   2. data.xml

      no breakpoint hit

Code

using  System;

using  Android.App;
using  Android.Content;
using  Android.Runtime;
using  Android.Views;
using  Android.Widget;
using  Android.OS;

namespace  HolisticWare.GoNIX.EXE_PL_924ClntMob_MonoDroid_netfx35
{
	[Activity(Label ="NavigationTest", MainLauncher =true)]
	public  class  MainPagewWelcome  :Activity
	{
		private  Button  buttonData;
		private  Button  buttonExceptionThrow;
		private  Button  buttonBackData;
		private  Button  buttonBackException;

		protected  override  void  OnCreate(Bundle  bundle)
		{
			base.OnCreate(bundle);

			// Set our view from the "main" layout resource
			SetContentView(Resource.Layout.mainpagewelcome);

			// Get our button from the layout resource,
			// and attach an event to it
			buttonData = FindViewById<Button>(Resource.Id.button_data);

			buttonExceptionThrow = FindViewById<Button>(Resource.Id.button_exceptiontest);
			buttonBackData = FindViewById<Button>(Resource.Id.button_bck1);
			buttonBackException = FindViewById<Button>(Resource.Id.button_back_exception);

			buttonData.Click +=new  EventHandler(buttonData_Click)
				//delegate { button.Text = string.Format("{0} clicks!", count++); }
				;

			buttonExceptionThrow.Click +=new  EventHandler(buttonExceptionThrow_Click);
			
			//buttonBackData.Click +=new EventHandler(buttonBackData_Click);
			//buttonBackException.Click += new EventHandler(buttonBackException_Click);
		}

		void  buttonBackException_Click(object  sender,EventArgs  e)
		{
			SetContentView(Resource.Layout.mainpagewelcome);
		}

		void  buttonBackData_Click(object  sender,EventArgs  e)
		{
			SetContentView(Resource.Layout.mainpagewelcome);
		}

		void  buttonExceptionThrow_Click(object  sender,EventArgs  e)
		{
			SetContentView(Resource.Layout.exceptionpage);

			return;
		}

		void  buttonData_Click(object  sender,EventArgs  e)
		{
			SetContentView(Resource.Layout.data);

			return;
		}
	}
}
