I've been trying to do a custom implementation of ISpinnerAdapter so I can
get an editable spinner. Following some examples online, I tried:

public class CompanyAdapter : ArrayAdapter<string> , ISpinnerAdapter
        {
            public CompanyAdapter(IntPtr ptr)
                : base(ptr)
            {
            }

            public override View GetView(int position, View convertView,
ViewGroup parent)
            {
                TextView text;
                if (convertView == null || !(convertView is TextView))
                {
                    text = new
TextView((Context)PageController.Instance.PageContext);
                    text.SetBackgroundColor(Android.Graphics.Color.Azure);
                }
                else
                {
                    text = convertView as TextView;
                }

                text.Text = this.GetItem(position).ToString();

                return text;
            }
        }


But no matter what I do (I've tried inheriting from BaseAdapter as well), I
keep getting a compile error that reads ""monodroid.exe" exited with code
1." I'm not sure how to get any more detail on it. Am I doing this the wrong
way?
