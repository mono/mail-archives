I'm calculating the distance between two Vector2, the calculation seems 
to be correct
until I'm calling ".ToString()" on any Vector2 and calling the .Distance 
function again.

I can't explain this behaviour (on MonoTouch the Vector2 class is 
working as intended)

I attached the OnCreate function and the Vector2 class, it isn't

The output produced is:
02-18 04:08:45.686  3558  3558 D Distance before ToString: 100,561
02-18 04:08:45.686  3558  3558 D Distance after ToString: 265693,5

         protected override void OnCreate(Bundle bundle)
         {
             base.OnCreate(bundle);
             Vector2 a = new Vector2(1, 1);
             Vector2 b = new Vector2(100, 100);
             float res = Vector2.Distance(a, b);
             Log.Debug("Distance before ToString", res.ToString());
             string s = b.ToString() + a;
             res = Vector2.Distance(a, b);
             Log.Debug("Distance after ToString", res.ToString());
         }

     public struct Vector2
     {
         public float X;
         public float Y;

         public Vector2(float x, float y)
         {
             this.X = x;
             this.Y = y;
         }

         public static float Distance(Vector2 value1, Vector2 value2)
         {
             float v1 = value1.X - value2.X, v2 = value1.Y - value2.Y;
             return (float)Math.Sqrt((v1 * v1) + (v2 * v2));
         }
     }
