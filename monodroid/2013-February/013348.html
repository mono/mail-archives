my saving to xml file code is here :
using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.IO;
using System.Xml;
using System.Text;

using Android.App;
using Android.Content;
using Android.OS;
using Android.Runtime;
using Android.Views;
using Android.Widget;

namespace EtSpSqlite.Resources
{
    [Activity(Label = "My Activity",MainLauncher=true)]
    public class XmlOne : Activity
    {
        Button btsave;
        Button btretreive;
        EditText eteno, etename, eteadd;
        TextView textename, texteadd;
        protected override void OnCreate(Bundle bundle)
        {
            base.OnCreate(bundle);
           
            // Create your application here
            SetContentView(Resource.Layout.XmlOne);
            eteno = FindViewById<EditText>(Resource.Id.eteno);
            etename = FindViewById<EditText>(Resource.Id.etename);
            eteadd = FindViewById<EditText>(Resource.Id.eteadd);
            btsave = FindViewById<Button>(Resource.Id.Btsave);
            btsave.Click += delegate { saveToXml(eteno.Text, etename.Text,
eteadd.Text); };
            btretreive = FindViewById<Button>(Resource.Id.BtRetreive);
            btretreive.Click += delegate { retFromXml(eteno.Text); };
        }
        private void saveToXml(string text1, string text2,string text3)
        {
            string path =
System.IO.Path.Combine(System.Environment.GetFolderPath
            (System.Environment.SpecialFolder.MyDocuments),
"myXMLfileOne.xml");
            try
            {
                XmlTextWriter textWriter = new XmlTextWriter(path, null);
                textWriter.WriteStartDocument();
                textWriter.WriteStartElement("Eno");
                textWriter.WriteStartElement(text1);
                textWriter.WriteEndElement();
                textWriter.WriteStartElement("Ename");
                textWriter.WriteStartElement(text2);
                textWriter.WriteEndElement();
                textWriter.WriteStartElement("Eadd");
                textWriter.WriteStartElement(text3);
                textWriter.WriteEndElement();
                textWriter.WriteEndDocument();
                textWriter.Close();
                Toast.MakeText(this, "saved successfully",
ToastLength.Long).Show();
                etename.Text = eteadd.Text = "";
            }
            catch (IOException ex)
            {
                Toast.MakeText(this, ex.Message, ToastLength.Long).Show();

                //Console.WriteLine("Exception thrown : {0}", ex.Message);
            }
        }
}
}
