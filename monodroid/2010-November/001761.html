Hello,

Is there a C# equivalent for the following code using anonymous types?

Uri uri = this.getIntent().getData();
this.getContentResolver().registerContentObserver(uri,
        true,
        new ContentObserver(new Handler()) {
            @Override
            public void onChange(boolean selfChange) {
                super.onChange(selfChange);
                if (MyViewer.this.cursor != null
                        && MyViewer.this.cursor.requery()) {
                    MyViewer.this.doSomething();
                }
            }
        });

The best I could come up with was this:

Android.Net.Uri uri = this.Intent.Data;
this.ContentResolver.RegisterContentObserver(uri, true,
        new MyContentObserver(new Handler(), this));

MyContentObserver is declared as:

class MyContentObserver : ContentObserver
{
    MyViewer viewer;

    public MyContentObserver(Handler handler, MyViewer viewer)
        : base(handler)
    {
        this.viewer = viewer;
    }

    public override void OnChange(bool selfChange)
    {
        base.OnChange(selfChange);
        if (this.viewer.cursor != null
            && this.viewer.cursor.Requery())
        {
            this.viewer.DoSomething();
        }
    }
}

Thanks,
Dan
