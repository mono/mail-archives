I extend the class Application on android and mark it with the attribute [Application], but 
the xamarin.android  do not make a register on  manifest file and  continue using the normal Application instance.


I’m target android 4.4, I’m using the last stable xamarin that are updated.


namespace CL.Android
{
    [Application]
    class CLApplication : Application
    {
        object objeto;

        public CLApplication (IntPtr handle, global::Android.Runtime.JniHandleOwnership transfer) : base (handle, transfer)
        {
            Current = this;
        }

        public static CLApplication Current { get; private set; }

        public void AtribuirObjeto<T> (T objeto)
        {
            this.objeto = objeto;
        }

        public T ObterObjeto<T> ()
        {
            return (T)objeto;
        }
    }
