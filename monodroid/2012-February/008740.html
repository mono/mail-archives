    public class CommandBasedClickableSpan : ClickableSpan
    {
        private readonly IMvxCommand _command;

        public CommandBasedClickableSpan(IMvxCommand command)
        {
            _command = command;
        }

        public override void OnClick(View widget)
        {
            _command.Execute();
        }
    }
