  {<br>  }</div><div>  public override object Target<br>  {<br>   get<br>   {<br>    var target = base.Target;</div><div>    if (target != null)<br>    {<br>     var javaObject = target as IJavaObject;</div><div>     if (javaObject != null)<br>
