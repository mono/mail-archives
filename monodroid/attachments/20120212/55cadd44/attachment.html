            <span style="color:blue">var</span> specSize = <span style="color:#2b91af">MeasureSpec</span>.GetSize(measureSpec);
 
 
            <span style="color:blue">if</span> (specMode == <span style="color:#2b91af">MeasureSpecMode</span>.Exactly)
            {
                <span style="color:green">// We were told how big to be</span>
                result = specSize;
            }
            <span style="color:blue">else</span>
            {
                result = _cellSize;
                <span style="color:blue">if</span> (specMode == <span style="color:#2b91af">MeasureSpecMode</span>.AtMost)
                {
                    <span style="color:green">// Respect AT_MOST value if that was what is called for by</span>
                    <span style="color:green">// measureSpec</span>
                    result = <span style="color:#2b91af">Math</span>.Min(result, specSize);
                }
            }
            <span style="color:blue">return</span> result;
        }

            <span style="color:blue">var</span> current = ActivePage;

                    <span style="color:blue">var</span> oval = <span style="color:blue">new</span> <span style="color:#2b91af">RectF</span>(x + 1, 1, x + _cellSize - 2, _cellSize - 2);
                    canvas.DrawOval(oval, _dotPaint);
                    canvas.DrawText(txt, x + (_cellSize - bounds.Width()) / 2, (_cellSize - _textHeight) / 2 + _ascent, _textPaint);
                }
                <span style="color:blue">else</span>
                {

                    canvas.DrawOval(oval, _dotPaint);
                }
            }
        }
