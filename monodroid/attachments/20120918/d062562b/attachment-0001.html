><br>
> 1.       Setup your project as specified in <a href="http://docs.xamarin.com/android/tutorials/Binding_a_Java_Library_%28.jar%29" target="_blank">http://docs.xamarin.com/android/tutorials/Binding_a_Java_Library_%28.jar%29</a> (I called it OsmDroid)<br>


> 2.       Add the following Jars:<br>
> a.       osmdroid-android-3.0.8 as the input jar<br>
> b.      httpmime-4.2.1 as reference jar<br>
> c.       osmdroid-third-party-3.0.8 as reference jar<br>
> d.      slf4j-android-1.5.8 as reference jar<br>
> 3.       In EnumMethods.xml you will want the following:<br>
><br>
>   <mapping jni-interface="org/osmdroid/api/IMapView"><br>
>     <method jni-name="setBackgroundColor" parameter="p0" clr-enum-type="Android.Graphics.Color" /><br>
>   </mapping><br>
><br>
>   <mapping jni-interface="org/osmdroid/api/IMyLocationOverlay"><br>
>     <method jni-name="onStatusChanged" parameter="p1" clr-enum-type="Android.Locations.Availability" /><br>
>   </mapping><br>
><br>
> 4.       Metadata.xml:<br>
><br>
>   <attr path="/api/package[@name='org.osmdroid.tileprovider.modules']/class[@name='MapTileModuleProviderBase.TileLoader']" name="visibility"><br>
>     public<br>
>   </attr><br>
>   <attr path="/api/package[@name='org.osmdroid.tileprovider.modules']/class[@name='MapTileModuleProviderBase.TileLoader']/method[@name='loadTile']" name="visibility"><br>


>     public<br>
>   </attr><br>
><br>
>   <attr path="/api/package[@name='org.osmdroid.tileprovider']/class[@name='MapTileProviderBase.ScaleTileLooper']/method[@name='handleTile']" name="visibility"><br>
>     public<br>
>   </attr><br>
>   <attr path="/api/package[@name='org.osmdroid.tileprovider']/class[@name='MapTileProviderBase.ZoomInTileLooper']/method[@name='handleTile']" name="visibility"><br>
>     public<br>
>   </attr><br>
>   <attr path="/api/package[@name='org.osmdroid.tileprovider']/class[@name='MapTileProviderBase.ZoomOutTileLooper']/method[@name='handleTile']" name="visibility"><br>


>     public<br>
>   </attr><br>
><br>
><br>
>   <!-- Draw() seems to be public for android so we need to be consistent about it.  C# doesn't like to mix visibility during inheritance. --><br>
>   <attr path="/api/package[@name='org.osmdroid.views.overlay']/class[@name='Overlay']/method[@name='draw']" name="visibility"><br>
>     public<br>
>   </attr><br>
>   <attr path="/api/package[@name='org.osmdroid.views.overlay']/class[@name='TilesOverlay']/method[@name='draw']" name="visibility"><br>
>     public<br>
>   </attr><br>

>     public<br>
>   </attr><br>
><br>
>   <remove-node path="/api/package[@name='org.osmdroid.tileprovider.tilesource']/class[@name='CloudmadeTileSource']" /><br>
><br>
>   <!-- Return IMapController in MapView.Controller so that we match IMapView. --><br>
>   <attr path="/api/package[@name='org.osmdroid.views']/class[@name='MapView']/method[@name='getController']" name="managedReturn"><br>
>     Org.Osmdroid.Api.IMapController<br>
>   </attr><br>
><br>

>     MapView.ProjectionImpl<br>

><br>

>   </attr><br>
