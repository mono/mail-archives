    Just few days ago did AdMobSample from samples:<br>
    Few things:<br>
    <ul>
      <li>set BuildAction for jar file to AndroidJavaLibrary (mine was
        none)</li>
      <li>set PublisherID in AndroidManifest.xml</li>
    </ul>
    Good post about JNI and VS<br>
<a class="moz-txt-link-freetext" href="http://manski.net/2012/03/30/accessing-java-classes-from-mono-for-android-via-jni/">http://manski.net/2012/03/30/accessing-java-classes-from-mono-for-android-via-jni/</a><br>
    <br>
    This should make AdMobSample compile and deploy. <br>
    <br>
    <title>Snippet</title>
    Mainfest:<br>
    <manifest
    xmlns:android=<a class="moz-txt-link-rfc2396E" href="http://schemas.android.com/apk/res/android">"http://schemas.android.com/apk/res/android"</a>
    android:installLocation="internalOnly"
    package="samples.admobtest"><br>
      <application><br>
        <meta-data android:value="your-key"
    android:name="ADMOB_PUBLISHER_ID" /><br>
        <activity android:name="com.admob.android.ads.AdMobActivity"
    android:theme="@android:style/Theme.NoTitleBar.Fullscreen"
    android:configChanges="orientation|keyboard|keyboardHidden" /><br>
        <receiver
    android:name="com.admob.android.ads.analytics.InstallReceiver"
    android:exported="true"><br>
          <intent-filter><br>
            <action
    android:name="com.android.vending.INSTALL_REFERRER" /><br>
          </intent-filter><br>
        </receiver><br>
      </application><br>
      <uses-sdk android:minSdkVersion="8"
    android:targetSdkVersion="13" /><br>
      <uses-permission android:name="android.permission.INTERNET"
    /><br>
    </manifest><br>
    <br>
    <br>
    Main.axml<br>
    <br>
    <title>Snippet</title>
    <?xml version="1.0" encoding="utf-8"?><br>
    <LinearLayout
    xmlns:android=<a class="moz-txt-link-rfc2396E" href="http://schemas.android.com/apk/res/android">"http://schemas.android.com/apk/res/android"</a><br>
        android:orientation="vertical"<br>
        android:layout_width="fill_parent"<br>
        android:layout_height="fill_parent"><br>
        <com.admob.android.ads.AdView<br>
            android:id="@+id/Ad"<br>
            android:layout_width="fill_parent"<br>
            android:layout_height="wrap_content" /><br>
        <Button<br>
            android:id="@+id/RefreshAd"<br>
            android:text="Refresh ad"<br>
            android:layout_width="wrap_content"<br>
            android:layout_height="wrap_content"/><br>
    </LinearLayout><br>
