class MyAnimationListener : Java.Lang.Object, 
        Android.Views.Animations.Animation.IAnimationListener 
    { 
        private Activity1 activity; 

        public MyAnimationListener(Activity1 activity) 
        { 
            // TODO: Complete member initialization 
            this.activity = activity; 
        } 

        public void OnAnimationEnd(Android.Views.Animations.Animation
animation) 
        { 

        } 

        public void OnAnimationRepeat(Android.Views.Animations.Animation
animation) 
        { 

        } 

        public void OnAnimationStart(Android.Views.Animations.Animation
animation) 
        { 

        } 
    } 

and after that..... 

animation.SetAnimationListener(new MyAnimationListener(this));
