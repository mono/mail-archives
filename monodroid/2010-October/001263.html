Hello, I am trying to create an application similar to the RSS feed 
example I created awhile back based on Scott Guthrie's example at 
http://weblogs.asp.net/scottgu/archive/2007/08/07/using-linq-to-xml-and-how-to-build-a-custom-rss-feed-reader-with-it.aspx.   
But when I run the application and enter something into the search 
EditText control and press the button the application waits for a few 
seconds then just closes with no error or message.  The code I used is 
the following.  I am wondering if this is a bug or if it is not yet 
supported in Monodroid.



using System;

using System.Linq;

using System.Xml.Linq;

using Android.App;

using Android.Content;

using Android.Runtime;

using Android.Views;

using Android.Widget;

using Android.OS;

using System.Net;

namespace FlckrFeed

{

[Activity(Label = "Digg Feeds", MainLauncher = true)]

public class Activity1 : Activity

{

public Activity1(IntPtr handle)

: base(handle)

{

}

protected override void OnCreate(Bundle bundle)

{

base.OnCreate(bundle);

// Set our view from the "main" layout resource

SetContentView(Resource.layout.main);

Button btnClick = FindViewById<Button>(Resource.id.myButton);

EditText txtTopic = FindViewById<EditText>(Resource.id.txtTopic);

btnClick.Click += delegate

{

string topic = (string)txtTopic.Text;

string diggUrl = 
String.Format("http://services.digg.com/stories/topic/{0}?count=20&appkey=http%3A%2F%2Fscottgu.com", 
topic);

WebClient diggService = new WebClient();

diggService.DownloadStringCompleted += new 
DownloadStringCompletedEventHandler(diggService_DownloadStringCompleted);

diggService.DownloadStringAsync(new Uri(diggUrl));

};

}

void diggService_DownloadStringCompleted(object sender, 
DownloadStringCompletedEventArgs e)

{

if (e.Error == null)

{

DisplayStories(e.Result);

}

else

{

Console.Write(e.ToString());

}

}

private void DisplayStories(string xmlContent)

{

XDocument xmlStories = XDocument.Parse(xmlContent);

var stories = from story in xmlStories.Descendants("story")

where story.Element("thumbnail") != null

//!story.Element("thumbnail").Attribute("src").Value.EndsWith(".gif")

select new DiggStory

{

Id = (int)story.Attribute("id"),

Title = ((string)story.Element("title")).Trim(),

ThumbNail = (string)story.Element("thumbnail").Attribute("src").Value,

// Description = ((string)story.Element("description")).Trim(),

HrefLink = new Uri((string)story.Attribute("link")),

NumDiggs = (int)story.Attribute("diggs"),

UserName = (string)story.Element("user").Attribute("name").Value,

};

ListView lstDiggs = FindViewById<ListView>(Resource.id.lstDiggs);

foreach(DiggStory digg in stories)

{
//   Temporary till I found out how to handle displaying the results in 
a ListView or GridView

Console.WriteLine(digg.Title);

}

}

}

}
