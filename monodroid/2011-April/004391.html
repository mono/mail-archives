John: He is an example of creating and reading the XML file based on your
code. I wrote/read from the SDCARD because it was where I had full
permissions without any special changes. I hope this helps. 

 

 

 

try { 

                           

       // CREATE XML DOCUMENT

                           

       System.Xml.XmlDocument xmlDocument = new System.Xml.XmlDocument ();

       

       System.Xml.XmlDeclaration xmlDeclaration =
xmlDocument.CreateXmlDeclaration ("1.0", "utf-8", String.Empty);

       

       xmlDocument.InsertBefore (xmlDeclaration,
xmlDocument.DocumentElement);

                           

                           

       System.Xml.XmlElement dataRoot = xmlDocument.CreateElement
("dataroot");

                           

       dataRoot.SetAttribute ("generated", DateTime.Now.ToString ());

                           

       xmlDocument.AppendChild (dataRoot);

                           

                           

       System.Xml.XmlElement jnyElement = xmlDocument.CreateElement ("JNY");

                           

       dataRoot.AppendChild (jnyElement);

                           

                           

       System.Xml.XmlElement uiElement = xmlDocument.CreateElement ("UI");

                           

       uiElement.InnerText = "9898989";

                           

       jnyElement.AppendChild (uiElement);

                           

                           

       // SAVE XML DOCUMENT

                           

       System.IO.StreamWriter streamWriter = new System.IO.StreamWriter
("/sdcard/myfile.xml", false);

                                  

       xmlDocument.Save (streamWriter);

                           

       streamWriter.Flush ();

                           

       streamWriter.Close ();

 

}

                     

catch (Exception exception) {

                     

       Android.Util.Log.Info ("Exception", exception.Message);

                           

}

                     

                     

try {

                           

       // READ XML FILE

                     

       System.IO.StreamReader streamReader = new System.IO.StreamReader
("/sdcard/myfile.xml");

                           

       System.Xml.XmlDocument document = new System.Xml.XmlDocument ();

                           

       document.Load (streamReader);

                           

       foreach (System.Xml.XmlNode currentNode in document.ChildNodes) {

                                  

              Android.Util.Log.Info ("Node", currentNode.InnerText);

                                  

       }                          

                           

}

                     

catch (Exception exception) {

                     

       Android.Util.Log.Info ("Exception", exception.Message);

                           

}
