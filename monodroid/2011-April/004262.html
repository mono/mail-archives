If you're using WebClient, you can add cookie functionality this way:

    public class CookieAwareWebClient : WebClient
    {

        private CookieContainer _container = new CookieContainer();
        public CookieContainer Cookies { get { return _container; } }

        protected override WebRequest GetWebRequest(Uri address)
        {
            WebRequest request = base.GetWebRequest(address);
            if (request is HttpWebRequest)
                ((HttpWebRequest)request).CookieContainer = _container;
            return request;
        }
    }

Then use the Cookies property to manage your cookies.

Hope this helps.
