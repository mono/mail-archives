using System;

using Android.App;
using Android.Content;
using Android.Runtime;
using Android.Views;
using Android.Widget;
using Android.OS;
using System.Text;

namespace HelloMonodroid
{
    [Activity(Label = "HelloMonodroid", MainLauncher = true, Icon =
"@drawable/icon")]
    public class Activity1 : Activity
    {

        protected override void OnCreate(Bundle bundle)
        {
            base.OnCreate(bundle);
            var tv = new TextView(this);
            tv.Text = "Hello, World Again";
            using (deviceWS.DeviceSettingsWS ws = new
deviceWS.DeviceSettingsWS())
            {
                using (System.Data.DataSet ds =
ws.GetSystemDateTime(System.Environment.MachineName, "TOKEN"))
                {
                    if (ds != null)
                    {
                        StringBuilder s = new StringBuilder();
                        for (int i = 0; i < ds.Tables.Count; i++)
                        {
                            s.Append(ds.Tables[i].TableName + ',');
                        }
                        tv.Text = s.ToString();
                    }
                }
            }

            SetContentView(tv);

        }
    }
}
