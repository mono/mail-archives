Sometimes I get an "Object reference not set to an instance of an object."  I've added the internet option to the AndroidManifest.xml file.  Any suggestions are appreciated.
 
protected override void OnCreate(Bundle bundle)
{
base.OnCreate(bundle);
// Set our view from the "main" layout resource
SetContentView(Resource.layout.main);
// Get our button from the layout resource,
// and attach an event to it
Button button = FindViewById<Button>(Resource.id.myButton);
button.Click += new EventHandler(button_Click);
}
void button_Click(object sender, EventArgs e)
{
Android.Media.MediaPlayer mp = Android.Media.MediaPlayer.Create(this,
Android.Net.Uri.Parse("http://morewally.com/cs/files/folders/1950/download.aspx"));
try
{
mp.Prepare();
mp.Start();
}
catch (System.Exception sysExc)
{
System.Exception se = sysExc;
while (se != null)
{
Android.Util.Log.E("graphics", sysExc.Message);
se = se.InnerException;
}
}
finally
{
mp.Release();
}
}
