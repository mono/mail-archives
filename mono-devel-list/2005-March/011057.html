class MemoryChecker
{
	public const int MB = 1000000;

	[STAThread]
	static void Main(string[] args)
	{
		Console.WriteLine("Pointer size(in bytes): {0}",
IntPtr.Size);

		byte[][] bigArray = new byte[10000][];

		int index = 0;
		while(true)
		{
			try
			{
				bigArray[index++] = new byte[MB];
			}
			catch(Exception e)
			{
				Console.WriteLine("Max Mb block allocated:
{0}", index);
				Console.WriteLine(e);
				break;
			}

		}
	}
}
