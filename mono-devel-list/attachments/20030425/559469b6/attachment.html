Hi,<BR>
<BR>
I've discovered a bug in the ProviderSectionHandler when using MS.NET FrameWork 1.1.<BR>
<BR>
In the method Create, it was never tested wether the object found in section.ChildNodes was a WhiteSpace or an Element, <BR>
so I ran into some nasty exceptions...<BR>
<BR>
<TT>        public virtual object Create(object parent,object configContext, XmlNode section)<BR>
                {<BR>
                        ProviderCollection providers=new ProviderCollection();<BR>
                        <FONT COLOR="#f80000">foreach (XmlElement ProviderNode in section.ChildNodes)</FONT><BR>
                        {<BR>
                                Provider provider=new Provider(<BR>
                                        GetStringValue(ProviderNode,&quot;name&quot;,true),<BR>
                                        GetStringValue(ProviderNode,&quot;connection&quot;,true),<BR>
                                        GetStringValue(ProviderNode,&quot;adapter&quot;,true),<BR>
                                        GetStringValue(ProviderNode,&quot;command&quot;,true),<BR>
                                        GetStringValue(ProviderNode,&quot;assembly&quot;,true),<BR>
                                        GetStringValue(ProviderNode,&quot;description&quot;,false));<BR>
                                providers.Add(provider);<BR>
                        }<BR>
                        return providers;<BR>
                }</TT><BR>
<BR>
So, I've made the following fix, and it works (changes are in red)<BR>
<BR>
<TT>        public virtual object Create(object parent,object configContext, XmlNode section)<BR>
                {<BR>
                        ProviderCollection providers=new ProviderCollection();<BR>
                        foreach (XmlElement ProviderNode in section.ChildNodes)<BR>
                        {<BR>
<FONT COLOR="#f80000">&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp; //Testing if it's an element and not a WhiteSpace<BR>
&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp; if (ProviderNode.NodeType == XmlNodeType.Element)<BR>
&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp; {</FONT><BR>
&nbsp;&nbsp;&nbsp;                                 &nbsp;&nbsp;&nbsp; Provider provider=new Provider(<BR>
&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;                                 GetStringValue(ProviderNode,&quot;name&quot;,true),<BR>
                                &nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp; GetStringValue(ProviderNode,&quot;connection&quot;,true),<BR>
&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;                                 GetStringValue(ProviderNode,&quot;adapter&quot;,true),<BR>
&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;                                 GetStringValue(ProviderNode,&quot;command&quot;,true),<BR>
&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;                                 GetStringValue(ProviderNode,&quot;assembly&quot;,true),<BR>
&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;                                 GetStringValue(ProviderNode,&quot;description&quot;,false));<BR>
&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;                                 providers.Add(provider);<BR>
&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp; <FONT COLOR="#f80000">}</FONT><BR>
                        }<BR>
                        return providers;<BR>
                }</TT><BR>
<BR>
I have attached the modified file.<BR>
