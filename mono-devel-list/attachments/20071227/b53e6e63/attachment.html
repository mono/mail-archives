<DIV>
<DIV>HttpListener&nbsp;listener = new HttpListener();</DIV>
<DIV>listener.Prefixes.Add(connectionEndPoint);</DIV>
<DIV>
<DIV>listener.Start ();</DIV>
<DIV>AsyncCallback callback =&nbsp;delegate(IAsyncResult ar) {</DIV>
<DIV><SPAN style="WHITE-SPACE: pre"></SPAN>HttpListenerContext&nbsp;httpContext =&nbsp;listener.EndGetContext(ar);</DIV>

<DIV>};</DIV>

<DIV>using(Stream stream = httpRequest.GetRequestStream()) {</DIV>
<DIV><SPAN style="WHITE-SPACE: pre"></SPAN>stream.Write(data, 0, date.Length);</DIV>
