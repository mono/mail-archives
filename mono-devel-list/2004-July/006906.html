>using System;
>
>class FlagsAttributeDemo
>{

>        Red = 1,
>        Green = 2,
>        Blue = 4
>    };
>
>    // Define an Enum with FlagsAttribute.

>    };
>
>    static void Main( )
>    {
>        Console.WriteLine( 
>            "This example of the FlagsAttribute attribute \n" +

>            "Enum without FlagsAttribute:\n" );
>        
>        // Display all possible combinations of values.
>        for( int val = 0; val <= 8; val++ )
