> 	xmlns:SOAP-ENC="http://schemas.xmlsoap.org/soap/encoding/"  
> 	xmlns:si="http://soapinterop.org/xsd">
>   <SOAP-ENV:Body>
>     <addpersonResponse>
> 	  <return xsi:type="xsd:string">Ok,1066295083-Hans-Meier</return>
