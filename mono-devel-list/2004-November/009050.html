You probably just need to replace the current 

  echo "PASS: $*" > $@ ; \
  rm -f $$testlogfile ; \

with

  if test -f $*.xml then \
    if $(XMLDOCDIFF) ../$*-ref.xml $*.xml >> $$testlogfile; then \
      echo "PASS: $*" > $@ ; \
      rm -f $*.xml $$testlogfile ; \
    else
      echo "FAIL: $*: xml comparison" > $@ ; \
  else \
    echo "PASS: $*" > $@ ; \
    rm -f $$testlogfile ; \
  fi
