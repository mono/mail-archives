> Hi,

> 
> void runTest(string filename)
> {
>     XmlDocument doc = new XmlDocument();
>     XmlReader xmlReader = XmlReader.Create(filename);
> 
>     while (!xmlReader.EOF)
>     {
> 	if ((xmlReader.NodeType == XmlNodeType.Element) &&
> (xmlReader.Depth == 1))
> 	    doc.ReadNode(xmlReader);
> 	else
> 	    xmlReader.Read();
>     }

> </RootElem>
> 
>
