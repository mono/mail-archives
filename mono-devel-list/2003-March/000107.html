Hi,

it's done like this:

===
using System;

public struct SPoint
{
  public int x,y;
  public SPoint (int x, int y) {
    this.x = x; this.y = y;
  }
};

public class CMain
{
  public static void Main()
  {
    SPoint[] List = new SPoint[] { new SPoint (1,2), new SPoint (3,4) };
  }
}
====
