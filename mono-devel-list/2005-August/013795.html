My thread code looks like this

public void run()
{
  while (true)
  {
    try
    {
       Monitor.Enter(this);
       t1.Text = (++count1).ToString();
       t2.Text = (++count2).ToString();
    }
    finally
    {
       Monitor.Exit(this);
    }
    Thread.Sleep(500);
  }
}

MethodInvoker mi = new MethodInvoker(this, Monitor.Enter);
while (true)
{
   this.BeginInvoke(mi);

   t2.Text = (++count2).ToString();
   Thread.Sleep(500);
}
