// bug 00002 - seg fault

using System;

    public struct x {
        protected byte p;

        public x(byte i) {
                p=i;
        }

        public byte P {
                get {
                        return P;
                }
        }
    }

    public static void Main() {
        x v = new x(3);
        Console.WriteLine("\tP {0}", v.P );
    }
}

(of course) the correct program, it runs fine.  Shouldn't there be a better
