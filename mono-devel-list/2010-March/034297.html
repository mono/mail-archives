MS started throwing with 2.0. It didn't throw before, though.

using System;
using System.Threading;

class Test
{
	static object locker = new object ();
	
	static void Main ()
	{
		// throws starting with MS.NET 4.0:
		Monitor.Enter (locker);
		Monitor.Exit (locker);
		Monitor.Exit (locker);

		// throws starting with MS.NET 2.0:
		Monitor.Exit (locker);
		Monitor.Exit (locker);		
	}
}
