I was able to shorten the C part of my example to almost nothing:

// monotest.cpp
#include <thread>
#include <mono/jit/jit.h>

void Thread() {
    MonoDomain* domain = mono_jit_init("monotest.dll");
    mono_jit_cleanup(domain);
}

int main() {
    //Thread();
    std::thread t(Thread);
    t.join();
}
