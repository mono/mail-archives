+Summary: Setting Control.Visible = true affects it's Position
+
+Using mono 1.1.16, when setting Control.Visible = true (if previously set
+to False), makes the control shift to the left and down a little
+(relatively to it's supposed and set position).
+
+Reproduce code:
+using System;
+using System.Collections.Generic;
+using System.Windows.Forms;
+
+namespace MonoTest13
+{
+    public class Form1 : Form
+    {
+        public Form1()
+        {
+            Button B1 = new Button();
+            B1.Visible = false;
+            B1.Top = 10;
+            B1.Left = 0;
+            B1.Visible = true;
+            this.Controls.Add(B1);
+
+            Button B2 = new Button();
+            B2.Top = 40;
+            B2.Left = 0;
+            this.Controls.Add(B2);
+        }
+
+        static void Main()
+        {
+            Application.Run(new Form1());
+        }
+    }
+}
+
+Expected results:
+The buttons should be at the left side of the form, in a line.
+
+Actual results:
+The buttons overlap and their Left property apparently differs.
