Mono doesn't support co-/contravariant generic parameters.

The nemerle program:

using System;

public interface A[+T] {
  getObj () : T;
}

public class B[T] : A[T] {
  public getObj () : T {
    throw Exception ();
  }
}

public class M {
  public static Main () : void {
    def b : B[string] = B ();
    def a : A[object] = b;
    _ = a.getObj ();
  }
}
