using System;
using System.Diagnostics;

class Program
{
    static void Main (string[] args)
    {
        Stopwatch sw = new Stopwatch ();
        int value = 5;
        int res = 0;
        sw.Start ();
        for (int i = 0; i < 100000000; ++i) {
            res = value.GetHashCode ();
        }
        sw.Stop ();
        Console.WriteLine (res);
        Console.WriteLine (sw.ElapsedMilliseconds);
    }
}
