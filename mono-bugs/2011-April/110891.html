// The test comes from VS breaking changes doc

using System;
using System.Linq;
using System.Linq.Expressions;

class Program
{
    static void Main ()
    {
        Test<S> ();
    }

    static void Test<T> () where T : I
    {
        Expression<Func<T, int>> e = l => l.SetValue () + l.Value;
        var arg = default (T);
        Console.WriteLine (e.Compile () (arg));
    }
}

interface I
{
    int Value { get; }
    int SetValue ();
}

struct S : I
{
    int value;

    public int Value {
        get {
            return value;
        }
    }

    public int SetValue ()
    {
        value = 5;
        return 0;
    }
}
