class Program
{
    public static void Main()
    {
        int i1, i2, i3;

        // 'as' is not permitted in a constant expression.
        if (null as object == null)
            i1 = 1;

        // 'is' is not permitted in a constant expression.
        if (!(null is object))
            i2 = 1;

        int j4 = 0;
        if ((0 == (j4 & 0)) && (0 == (0 & j4)))
            i3 = 1;

        System.Console.WriteLine("{0}{1}{2}{3}{4}{5}", i1, i2, i3);
    }
}
