+
+public class M {
+  public static void Main() {
+    Uri uri = new Uri ("http://h/p?q=a%26b");
+    Console.WriteLine (uri.Query);
+    Console.WriteLine (uri.ToString());
+  }

+EOF
+2. mcs Uri.cs
+3. mono Uri.exe
