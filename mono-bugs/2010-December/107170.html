Description of Problem:

The code compiles well under .NET 3.5 but fails to compile under Mono 2.8 (OSX
running in VirtualBox). There is something wrong with resolving of correct
this[...] property in base interfaces.

To reproduce the problem compile the code below:

using System.Collections;
using System.Collections.Generic;

namespace MonoBug
{
    public interface IArray : IList
    {
        object this[params int[] index] { get; set; }
    }

    public interface IArray<T> : IArray, IList<T>
    {
        new T this[params int[] index] { get; set; }
    }

    public class Class1<T> : List<T>, IArray<T>
    {
        IArray<T> Source { get { return null; } }

        /// <summary>
        /// BUG: compiles on Windows but generates error under Mono (OSX)
        /// 
        /// error CS0584: Internal compiler error: Cannot cast from source type
to destination type.
        /// error CS0266: Cannot implicitly convert type `object' to `T'. An
explicit conversion exists (are you missing a cast?)
        /// </summary>
        /// <param name="index"></param>
        /// <returns></returns>
        T IList<T>.this[int index]
        {
            get { return Source[index]; } 
            set { Source[index] = value; }
        }

        object IArray.this[params int[] index] { get { return null; } set { } }

        public T this[params int[] index] { get { return default(T); } set { }
}
    }
}
