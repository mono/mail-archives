+using System;
+
+delegate void Do ();
+
+class T {
+        static void doit (int v) {
+                Console.WriteLine (v);
+        }
+        static void Main () {
+                Do[] arr = new Do [5];
+                for (int i = 0; i < 5; ++i) {
+                        arr [i] = delegate {doit (i);};
+                }
+                for (int i = 0; i < 5; ++i) {
+                        arr [i] ();
+                }
+                for (int i = 0; i < 5; ++i) {
+                        arr [i] = delegate {doit (i);};
+                }
+        }
+
+}
+
