+    public Stone Get(Position p) {
+        if (board.Contains(p)) {
+            return (Stone)board[p];
+        } else {
+            return Stone.None;
+        }
+    }

+    public Stone Get(int x, int y) {
+        foreach (Position p in board.Keys) {
+            if ((p.x == x) && (p.y == y)) {

+        return Stone.None;
