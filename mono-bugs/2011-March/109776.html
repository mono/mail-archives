public void Method_Condition_Result()
{
    try
    {
        Directory.CreateDirectory(@".\tmp\folder");

        //Expected: "./tmp/folder"
        //But was:  ".\\folder"

    }
    finally
    {
        Directory.Delete(@".\tmp\", true);
    }
