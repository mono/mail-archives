The following test fails

[Test]
public void BSTR_Roundtrip ()
{
        string s = "mono";
        IntPtr ptr = Marshal.StringToBSTR (s);
        string s2 = Marshal.PtrToStringBSTR (ptr);
        Assert.AreEqual (s, s2, "string");
}

        Strings differ at index 4.

        expected:<"mono">
