using System;
using System.IO;

public class test
{
        public static void Main ()
        {
                bool use_stream = false;
                StreamReader fs = new StreamReader ("/tmp/foo");
                using (use_stream ? fs : null) {
                        Console.WriteLine ("hello");
                }
        }
}

Compilation failed: 1 error(s), 0 warnings
