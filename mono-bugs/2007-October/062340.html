Steps to reproduce issue:

1. Run code
2. Close window by pressing cross in upper right window

Observed:

Window is closed

Expected:

Window should not closed. Lookup form should appear.


using System;
using System.Windows.Forms;

        public class Form1 : Form {
MyComboBox MyCombobox = new MyComboBox();

                [STAThread]
                public static void Main() {
                        try {
                                Application.Run(new Form1());
                        }
                        catch (Exception e) {
                                MessageBox.Show(e.ToString());
                        }
                }

                public Form1() {
                        this.Controls.Add(MyCombobox);
                TextBox         tb = new TextBox();
                        tb.Top = 200;
                        Controls.Add(tb);
                }

        }


public class MyComboBox : ComboBox {

        protected override void
OnValidating(System.ComponentModel.CancelEventArgs e) {
                e.Cancel = true;
                Form f = new Form();
                f.Text = "Modeless lookup form";
                f.Show();
        }
}
