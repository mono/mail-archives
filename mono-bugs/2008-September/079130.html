using System;
using System.Linq;
using System.Linq.Expressions;

class Program
{
    static void Main()
    {
        Test<S>();
    }

    static void Test<T>() where T : I
    {       
        Expression<Func<T, int>> e = x => x.SetX();
        e.Compile()(default (T)); 
    }
}

interface I
{
    int X { get; }
    int SetX();
}

struct S : I
{
    public int X { get; private set; }
    public int SetX()
    {
        X = 7;
        return 0;
    }
}


1. compile
2. run

Expected: no exception
