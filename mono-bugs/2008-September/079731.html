This piece of code:

public class MainProgram
{
    public static void Main()
    {
    }

    public void AnyMethod<T>()
    {
        Action<string> outerAction = outerActionParam =>
        {
            Action<string> innerAction = innerActionParam =>
innerActionParam.Contains(outerActionParam);
        };
    }
}
