Paolo, you forgot that arrays implement way more interfaces than just object.
The following piece of code doesn't throw an exception.


using System;
using System.Collections.Generic;

class A {}
class B : A{}

class C : A {
  static void Main ()
  {
    B [] aa = new B [1];
    IList<A> io = aa;
    io [0] = new C ();
    B a = aa [0];
    Console.WriteLine ("{0}", a.GetType ());
  }
}
