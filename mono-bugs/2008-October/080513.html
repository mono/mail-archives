MCS crashes with an internal server error in the "*c++ = (byte)(*a++ * *b++);"
line.

Using "*c++ = (byte)(*a++ * (*b++));" instead works just fine.

using System;




public struct Sample
{
        private byte a;

        public static unsafe Sample operator * Sample v1, Sample v2)

        {

                Sample res = new Sample ();
                byte *a = &v1.a;
                byte *b = &v2.a;
                byte *c = &res.a;
                for (int i = 0; i < 16; ++i)
                        *c++ = (byte)(*a++ * *b++);
                return res;

        }
}
