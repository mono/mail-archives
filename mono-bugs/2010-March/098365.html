What platform is this ? On x86 linux, structures are returned differently than
scalar values, so you have to write:

typedef struct {
   void *p;
} IntPtrStruct;

IntPtrStruct
jni4net_zero () {
   IntPtrStruct s;

   s.p = NULL;

   return s;
}
