+		public event EventHandler MouseClick;
+	}

+	{
+		public new event EventHandler MouseClick;
+	}
+	
+	public class C : B
+	{
+		public new void MouseClick ()
+		{
+			Console.WriteLine ("This should be printed");
+		}
+	}
+	
+	static public void Main ()
+	{
+		C myclass = new C ();
