+
+class Class1
+{
+	static int Main(string[] args)

+		string parse_me = "9999999999999999999999";

+		long val = 0;
+		char c;
+		
+		for (int i = 0; i < len; i++){
+			c = parse_me [i];
+
+			if (c >= '0' && c <= '9'){
+				val = checked (val * 10 + (c - '0'));
+			}
+		}
+		return 1;
+	}
