The follow snippet:

using System;

namespace Bug
{
        public class A : B
        {
                public A ()
                {
                        var dingus = new B ();

                        dingus.Event += delegate {
                                // Comment me and a valid ctor is generated
                                int prop = dingus.Prop;
                        };
                }
        }

        public class B
        {
                public B ()
                {
                }

                public int Prop { get; set; }

                public event EventHandler Event;
        }

        public delegate void EventHandler ();
}
