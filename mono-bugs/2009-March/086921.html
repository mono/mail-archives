I guess that I'm able to provide required information.

Here is sample program:

using System;
namespace test
{
    interface I<T>
    {
        T Get();
    }

    class G<A,C>
        where C:struct, I<A>
    {

        public void M<CA>(G<A, CA> g)
            where CA:struct, I<A>
        {
        }
    }

    struct Iint1:I<int>
    {
        public int Get()
        {
            return 0;
        }
    }

    struct Iint2:I<int>
    {
        public int Get()
        {
            return 0;
        }
    }

    class MainClass
    {
        public static void Main (string[] args)
        {
            G<int, Iint1> gint1 = new G<int, Iint1>();
            G<int, Iint2> gint2 = new G<int, Iint2>();
            gint1.M<Iint2>(gint2);
        }
    }
}
