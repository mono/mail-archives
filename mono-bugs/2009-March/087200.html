...
using (OracleDataReader reader = cmd.ExecuteReader())
{
  while(reader.Read())
  {
    //read all rows form first cursor

  if (reader.NextResult())
  {
    while(reader.Read())
    {
      //reader.Read() return false always!!!
    }
  }
}

Expected Results:  
Can read rows from second cursor
