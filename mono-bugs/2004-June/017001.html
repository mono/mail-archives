+using System.Globalization;
+
+public class Test
+{
+        public static void Main ()
+        {
+                DateTime.ParseExact ("02#25#2002", "MM/dd/yyyy",
+CultureInfo.InvariantCulture);
+//              DateTime.ParseExact ("02-25-2002", "MM/dd/yyyy",
+CultureInfo.InvariantCulture);
+        }

+variant of "MM/dd/yyyy".
