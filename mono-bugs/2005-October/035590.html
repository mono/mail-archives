+
+Request.PathInfo:
+mod_mono: /200309OeSTM/index.aspx
+iis5: /200309OeSTM/
+

+1. enter a url with PathInfo that ends in a slash 
+2. 
+3.

+/200309OeSTM/index.aspx

+/200309OeSTM/
+

+myserver.conf:
+<VirtualHost ipaddr:80>
+    ServerName myserver
