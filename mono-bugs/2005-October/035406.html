+

+
+class Base
+{
+        public virtual void Foo ()
+        {
+                Console.WriteLine ("Foo");
+        }
+        public virtual void Foo<T> ()
+        {
+                Console.WriteLine ("Foo<>");
+        }
+}
+
+class Derived : Base
+{
+        public override void Foo<T> ()
+        {
+                Console.WriteLine ("Derived.Foo<>");
+        }
+}
+
+class Driver
+{
+        public static void Main ()
+        {
+                new Derived ().Foo<int> ();
+        }
+}
