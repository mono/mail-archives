How about

static bool CheckDomain (string domain, string host)
{
    if (domain == "") {
        return false;
    } else if (domain[0] != ".") {
        domain = "." + domain;
    }
    if (host.Length != "" && host[0] != ".") {
        host = "." + host;
    }
    return host.EndsWith(domain);
}
