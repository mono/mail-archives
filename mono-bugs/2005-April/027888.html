+using System;
+using System.Security.Cryptography;
+
+class C {
+        static void Main ()
+        {
+                Type t = typeof (CipherMode);
+                Console.WriteLine (t);
+                Console.WriteLine (t.IsSerializable);
+                object [] objs = t.GetCustomAttributes (false);
+                foreach (object o in objs)
+                        Console.WriteLine (o.GetType ());
+        }
