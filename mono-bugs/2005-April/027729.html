+using System;
+using System.Collections;
+
+class CompileBug
+{
+	struct PassStruct
+	{
+		public float t;
+	};
+
+	static void Main ()
+	{
+		PassStruct test;
+		test.t = 0F;
+		DoIterator (test);
+	}
+	
+	static IEnumerator DoIterator (PassStruct test)
+	{
+		yield return test.t;
+	}
+};
