To reproduce, compile and run the following code:

using System;
using System.IO;

class Program
{
        static void Main ()
        {
                int? a = 5;
                int? b = 7;

                Console.WriteLine (AreEqual (a, b));
                Console.WriteLine (AreEqual (a, a));
        }

        static bool AreEqual (IComparable x, IComparable y)
        {
                return x.CompareTo (y) == 0;
        }
}
