+Summary: mcs doesn't see an inherited explicit interface member implementation

+The following code compiles fine with Microsoft's csc but produces
+an error with mcs.
+
+using System;
+public interface I
+{
+	void m ();
+}
+public interface II : I
+{
+	void mm ();
+}
+public class X : I
+{
+	void I.m () {}
+}
+class Y : X, II
+{
+	public void mm () {}
+	public static void Main (string[] args) {}
+}
+

+
