+Description of Problem:
+Deadlock in ReaderWriterLock.UpgradeToWriterLock()
+
+Steps to reproduce the problem:
+
+Run this code:
+===================================================================
+using System;
+using System.Threading;
+
+namespace Example {
+   class Class1 {
+      static ReaderWriterLock _rwl = new ReaderWriterLock();
+      const int _timeout = Timeout.Infinite;
+
+      [STAThread]
+      static void Main(string[] args) {
+         Thread th = new Thread(new ThreadStart(Thread2));
+         th.Start();
+         Thread.Sleep(1000);
+         Console.WriteLine("1 - AcquireReaderLock, before");
+         _rwl.AcquireReaderLock(Timeout.Infinite);
+         Console.WriteLine("1 - AcquireReaderLock, after");
+         try {
+            Console.WriteLine("1 - UpgradeToWriterLock, before");
+            LockCookie lc = _rwl.UpgradeToWriterLock(_timeout);
+            Console.WriteLine("1 - UpgradeToWriterLock, after");
+            Thread.Sleep(500);
+            Console.WriteLine("1 - Release, before");
+         }
+         finally { _rwl.ReleaseReaderLock(); }
+         Console.WriteLine("1 - End");
+      }
+
+      static void Thread2() {
+         Console.WriteLine("2 - AcquireReaderLock, before");
+         _rwl.AcquireReaderLock(Timeout.Infinite);
+         Console.WriteLine("2 - AcquireReaderLock, after");
+         try {
+            Thread.Sleep(3000);
+            Console.WriteLine("2 - UpgradeToWriterLock, before");
+            LockCookie lc = _rwl.UpgradeToWriterLock(_timeout);
+            Console.WriteLine("2 - UpgradeToWriterLock, after");
+            Thread.Sleep(2000);
+            Console.WriteLine("2 - Release, before");
+         }
+         finally { _rwl.ReleaseReaderLock(); }
+         Console.WriteLine("2 - End");
+      }
+   }
+}
+===================================================================
