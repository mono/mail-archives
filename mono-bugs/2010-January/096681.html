Test case:

>>>>>>>>>>>>>>>>>>>>>>>
using System;
using System.Text;

namespace test
{
    class MainClass
    {
        static void Main(string[] args)
        {
            var a = "\uae40\ud604\uc2dd";
            var b = a.Normalize(NormalizationForm.FormKD);
            print(a);
            print(b);
        }

        private static void print(string s)
        {
            foreach(var c in s)
            {
                Console.Write("{0:x} ", (int)c);
            }
            Console.WriteLine();
        }
    }
}
