Starting with trunk r88411 Nemerle doesn't work anymore.

public abstract class A[T]
{
    protected a : T
}

public abstract class B[T] : A[T]
{
}

public class C : B[int]
{
    public F() : int
    {
        a
    }

    public static Main(args : array[string]) : int
    {
        0
    }
