+class X {
+	static void Main ()
+	{
+		long x = 0;
+		for (long i = 0; i < 10000000; i ++) {
+			if (i == 2)
+				x = i * i;
+			else if (i % 2 != 0)
+				x = i * (i - 1);
+			else {
+				x = x * (i / 2);
+				x = x * x;
+			}
+		}
+	}
