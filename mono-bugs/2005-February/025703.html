+using System.Threading;
+
+public class test {
+ public static void StartNewThread() {
+  Thread t = new Thread( new ThreadStart( StartNewThread ) );
+  t.Start();
+  DateTime dt = DateTime.Now;
+ }
+
+ public static void Main( string[] args ) {
+  Thread t = new Thread( new ThreadStart( StartNewThread ) );
+  t.Start();
+  Thread.Sleep( 1000 );
+  Environment.Exit( 0 );
+ }
+}
+--8<-----------------------------------------
