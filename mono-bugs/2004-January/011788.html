+

+

+	public void P ()

+public class X {
+	public void PP ()
+	{}
+}
+
+File: 3.cs
+class Y {

+	static void Main ()
+	{
+		a.P ();

+mcs -target:library 1.cs
+mcs -target:library 2.cs
+mcs 3.cs -r:1.dll -r:2.dll
+mcs 3.cs -r:2.dll -r:1.dll
+
+Notice the difference in output in the last two.
