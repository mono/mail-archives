                Matrix matrix = new Matrix ();

                clipRegion.MakeInfinite ();

                Rectangle smaller = new Rectangle (5, 5, -10, -10);
                Rectangle bigger = new Rectangle (-5, -5, 12, 12);

                        Console.WriteLine (rect);
