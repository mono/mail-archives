{
    public class BaseClass
    {
        public static implicit operator string (BaseClass c)
        {
            return "I'm implicit";
        }
    }

    public static void Main (string [] args)
    {
        Console.WriteLine ("Equal? {0}", new BaseClass () == "I'm implicit");
    }
}
