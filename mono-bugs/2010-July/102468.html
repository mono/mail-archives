Example code:

var raw =
"multipart/mixed;boundary=\"----=_NextPart_000_00DA_01CB28E8.A3142040\"";
System.Net.Mime.ContentType contentType = new ContentType(raw);
Console.Write("Boundary:{0}", contentType.Boundary);
