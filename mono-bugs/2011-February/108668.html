Repro:

using System;
using System.Diagnostics;
class Zombie {
  static void Main () {
    Process.Start ("sleep", "1").Dispose ();
      Thread.Sleep (1000000);
      // now check using ps that the sleep process is a zombie process.
    }
}

This works:

using System;
using System.Diagnostics;
class Zombie {
  static void Main () {
    Process.Start ("sleep", "1").WaitForExit (1);
      Thread.Sleep (1000000);
      // now check using ps that the sleep process is a zombie process.
    }
}
