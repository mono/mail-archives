+The following testcase illustrates a bug in the flow analysis code.  The
+"WriteLine" line is considered unreachable and eliminated from the
+generated code.
+

+        static void Main ()
+        {
+                int i = 5;
+                switch (i) {
+                case 5:
+                        if (i == 5)
+                                break;
+                        return;
+                default:
+                        return;
+                }
+                System.Console.WriteLine (i);
+        }
+}
