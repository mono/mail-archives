   Type d1 = typeof(ServiceController<>);
   Type[] typeArgs = { typeof(T) };
   Type makeme = d1.MakeGenericType(typeArgs);
   ServiceController<T> serviceController =    
Activator.CreateInstance(makeme) as ServiceController<T>;
