The following bench is 3-4 times slower on x86 after the switch to the linear
IR.
using System;

class Program {
    public static void Main() {
      double d = 0;
      for (double i = 0; i < 100000000; i++) {
        d += i + 1000.0;
        d += i - 1000.0;
        d += i * 1000.0;
        d += i / 1000.0;
        d += i % 1000.0;
      }
      Console.WriteLine (d);
   }
}
