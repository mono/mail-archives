+Mono C# compiler (gmcs) reports an internal error when it tries to resolve
+what of two generic functions with the same name (but different signature)

+
+Steps to reproduce the problem:
+Try to compile the following code:
+=== test.cs ===
+public interface IFoo
+{
+        void Foo<T>(ref T? v) where T:struct;
+        void Foo<T>(ref T v) where T:new();
+}
+
+public struct Point
+{
+        int x, y;
+        public Point(int x, int y) { this.x = x; this.y = y; }
+}
+
+struct TestPoint
+{
+        public static void Serialize(IFoo h)
+        {
+                Point? point = new Point(1,2);
+                h.Foo(ref point);
+        }
+        public static void Main(){}
+}
+============
