+using System;
+using System.Windows.Forms;
+using System.Drawing;
+
+public class TestForm : Form
+{
+	static void Main ()
+	{
+		Application.Run (new TestForm ());
+	}
+
+	TextBox tb1;
+	TextBox tb2;
+	RadioButton rb1;
+	RadioButton rb2;
+
+	public TestForm ()
+	{
+		tb1 = new TextBox ();
+		tb1.Location = new Point (5, 5);
+		tb1.Text = "One";
+		tb1.Parent = this;
+
+		tb2 = new TextBox ();
+		tb2.Location = new Point (125, 5);
+		tb2.Text = "Two";
+		tb2.Parent = this;
+
+		rb1 = new RadioButton ();
+		rb1.Checked = true;
+		rb1.Text = "Focus One";
+		rb1.Location = new Point (5, 35);
+		rb1.Parent = this;
+		rb1.CheckedChanged += delegate (object o, EventArgs args)
+		{
+			if (rb1.Checked) // If activated, move focus to tb1
+				tb1.Focus ();
+		};
+
+		rb2 = new RadioButton ();
+		rb2.Checked = false;
+		rb2.Text = "Focus Two";
+		rb2.Location = new Point (125, 35);
+		rb2.Parent = this;
+		rb2.CheckedChanged += delegate (object o, EventArgs args)
+		{
+			if (rb2.Checked) // If activated, move focus to tb2
+				tb2.Focus ();
+		};
+	}
+}
