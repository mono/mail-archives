+

+

+
+using System;
+using System.IO;
+
+public class prueba
+{
+	public static void Main ()
+	{
+		byte [] b = new byte [30];

+		try {
+			BinaryReader r = new BinaryReader (m);
+			r.Close ();
+		} catch (Exception e) {
+			Console.WriteLine (e.ToString ());
+		}
+	}
