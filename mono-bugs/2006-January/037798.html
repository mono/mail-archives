+Compilation failed: 1 error(s), 0 warnings

+

+No compiler error.  Works with csc 2.0.
+
+

+
+// Program Bug20a.cs:
+using System.Collections.Generic;
+
+public class MyList<T> : List<T> { 
+  public void AddAll<U>(IEnumerable<U> xs) where U : T {
+    foreach (T x in xs) 
+      Add(x);
+  }

+using System;

+  }
