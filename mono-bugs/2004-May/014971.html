+OK, here is the original code that causes the problem. The same code
+compiles fine with Visual C#:
+
+public bool ice_invoke(string operation, OperationMode mode, byte[]
+inParams, out byte[] outParams,
+                               Context context) // Line 174
+        {
+            int __cnt = 0;
+            while(true)
+            {
+                try
+                {
+                    Object_Del __del = __getDelegate();
+                    return __del.ice_invoke(operation, mode, inParams,
+out outParams, context);
+                }
+                catch(IceInternal.NonRepeatable __ex)
+                {
+                    if(mode == OperationMode.Nonmutating || mode ==
+OperationMode.Idempotent)
+                    {
+                        __cnt = __handleException(__ex.get(), __cnt);
+                    }
+                    else
+                    {
+                        __rethrowException(__ex.get());
+                    }
+                }
+                catch(LocalException __ex)
+                {
+                    __cnt = __handleException(__ex, __cnt);
+                }
+            }
+        }
+

+
+ObjectPrxHelper.cs(174) error CS0177: The out parameter `outParams'

+number is out by one -- no big deal though.)
