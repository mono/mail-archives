+		public override bool IsEnabled {
+			get {
+				IWorkbenchWindow window   = WorkbenchSingleton.Workbench.ActiveWorkbenchWindow;
+				IEditable        editable = window != null ? window.ActiveViewContent as IEditable : null;
+				if (editable != null) {
+					return true;
+				}
+				return false;

+		}
