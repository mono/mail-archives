        {

            // Specify the directory you want to manipulate.

            string path = @"\home\MyDir";

            try 

            {

                // Determine whether the directory exists.

                if (Directory.Exists(path))

                    Console.WriteLine("That path exists already.");

                    Console.ReadLine();
