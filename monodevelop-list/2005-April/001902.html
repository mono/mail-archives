Quick little patch to add a --repair option to monodevelop.in to delete
the "~/.config/MonoDevelop/" folder's contents. 

Hope its useful.

 cd $MD_BIN_PATH
 
 case x$1 in
+	x--repair)
+		echo "** Removing Profile Data **"
+		rm -rfv ~/.config/MonoDevelop/
+		exit 0
+		;;
 	x--profile)
 		exec @RUNTIME@ --profile ./MonoDevelop.exe $@
 		exit 0
