public class Meals
{
  [Section("Dinner")]
  [Caption("Desert")]
  public Desert Desert { get; set; }

}

public enum Desert
{
  IceCream,
  MilkShake,
  ChocolateCake
}

you can do this:

ice_cream == "ice cream"
