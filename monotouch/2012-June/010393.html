Thanks Tomasz,

Here is a much tidier version, I have sized it for portrait view on the iPhone, it will display both the map and street view (map above the street view) with a search box which is also required for me as the search parameters may not return something so I need the user to be able to customise the search on screen.

What is important for me is if I have the lat and the long then start with that and show the map, if they don't exist then start the map at the most central point in london (trafalgar square), if there is no lat and long then the user can click search to try to get the location.

I needed a solution that gets round the apple/google issues that are up and coming I believe this is pretty future proof (famous last words!)

In the previous example I searched and replaced all the "" with %%1%% this makes it very easy for me to build HTML strings without having to inject speech marks everywhere, if someone has a better idea I would love to hear it.

Thanks

Alex


<script type="text/javascript" src="http://maps.googleapis.com/maps/api/js?sensor=false"></script>
 
<script type="text/javascript">
    var geocoder;
    var map;
    var address = '';
    var panorama;

    var panoramaOptions = {
        position: fenway,
            pov: {heading: 34,
                pitch: 10,
                zoom: 1}
    };

    function initialize() {
        geocoder = new google.maps.Geocoder();
        var latlng = new google.maps.LatLng(51.507526, -0.12795);
        var myOptions = {
            zoom: 18,
            center: latlng,
            mapTypeId: google.maps.MapTypeId.ROADMAP
        }
        map = new google.maps.Map(document.getElementById("map_canvas"), myOptions);    
        panorama = new google.maps.StreetViewPanorama(document.getElementById("panorama"),panoramaOptions);
        map.setStreetView(panorama);
    }

    function returnMapLocation() {
        var address = document.getElementById("address").value;
        geocoder.geocode( { 'address': address, 'region': 'uk'}, function(results, status) {
            if (status == google.maps.GeocoderStatus.OK) {
                map.setCenter(results[0].geometry.location);
                var marker = new google.maps.Marker({
                    map: map,
                    position: results[0].geometry.location
                });
        
                panorama = new google.maps.StreetViewPanorama(document.getElementById("panorama"),panoramaOptions);
                map.setStreetView(panorama);
        
            } else {
                alert("Could not find address: " + status);
            }
        });
    }
</script>
<body onload="initialize()">
  <div>
    <input id="address" type="textbox" value="london bridge station">
    <input type="button" value="Search" onclick="returnMapLocation()">
  </div>
        <div id="map_canvas" style="width: 320px; height: 300px"></div>
        <div id="panorama"   style="width: 320px; height: 300px"></div>
</body>
