On Fri, Jul 8, 2011 at 13:16, vbisbest <vbisbest at yahoo.com> wrote:
> The UISearchBar has an event that fires when a scope button changes.  Here
> are some snippets, "this" is a DialogViewController that has the SearchBar
> exposed per my post above:
>
>        public enum SearchScope
>        {
>                All,
>                Subject,
>                From,
>                Advanced
>        }
>
>                public override void ViewDidLoad ()
>                {
>                        this.SearchBar.SelectedScopeButtonIndexChanged += ScopeButtonChanged;
>                        base.ViewDidLoad ();
>                }
>
>                private void ScopeButtonChanged(object sender, System.EventArgs args)
>                {
>                        // Search scope has changed
>                        searchScope = (SearchScope)this.SearchBar.SelectedScopeButtonIndex;
>                }
