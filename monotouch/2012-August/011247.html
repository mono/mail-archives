    var root = new RootElement("foo");
    	  root.Add (new Section("foo", "bar") {
    	  new ActivityElement()
    	  {
    	    Caption = "foo",
    	    Animating = true
    	  }
    	});

    	var dvc = new DialogViewController(root, false);


    	window.RootViewController = dvc;
