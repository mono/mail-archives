        void ServiceAddressResolved (object sender, EventArgs e)
        {
            NSNetService ns = sender as NSNetService;

            if (ns != null) {
                foreach (NSData addressData in ns.Addresses) {

                    long l = addressData.Bytes.ToInt64 ();

                    long lHost = IPAddress.NetworkToHostOrder (l);

                    string ipAddress = new IPAddress (lHost).ToString ();

                    Console.WriteLine (ipAddress);
                }
            }
        }
