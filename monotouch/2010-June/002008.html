Hi Wayne,

Yeah, it's basically just passing in the PDF to the web view;


	var thePath = NSBundle.MainBundle.PathForResource("YourPDF", "pdf");

	if (thePath != null)
	{
		var url = NSUrl.FromFilename(thePath);
		var requestObj = NSUrlRequest.FromUrl(url);
		webView.LoadRequest(requestObj);
	}
