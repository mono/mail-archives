This in UIApplicationDelegate.FinishedLaunching  seems to correct it:

if (UIDevice.CurrentDevice.UserInterfaceIdiom == UIUserInterfaceIdiom.Phone)
{
    window.Frame = new RectangleF (0, 0, 320, 480);
} 
else if (UIDevice.CurrentDevice.UserInterfaceIdiom ==
UIUserInterfaceIdiom.Pad)
{
    window.Frame = new RectangleF (0, 0, 768, 1024);
}
