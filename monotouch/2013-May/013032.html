namespace ftrack2ios
{
     [Register ("HomeView")]
	partial class HomeView
     {
         [Outlet]
		MonoTouch.UIKit.UIBarButtonItem btnSettings { get; set; }

         [Outlet]
		MonoTouch.UIKit.UILabel lblTitle { get; set; }

         void ReleaseDesignerOutlets()
         {
             if (btnSettings != null)
             {
                 btnSettings.Dispose();
                 btnSettings = null;
             }

             if (lblTitle != null)
             {
                 lblTitle.Dispose();
                 lblTitle = null;
             }
         }
     }
