Hi Andreas,

try something like this;

public partial class TableViewController : UITableViewController
    {
        public TableViewController (IntPtr handle) : base (handle)
        {
        }

        public override void ViewDidLoad ()
        {
            base.ViewDidLoad ();
        
            // Swipe Gesture
            UISwipeGestureRecognizer gesture;
            gesture = new UISwipeGestureRecognizer (StartAlertOnSwipe);
            gesture.Direction = UISwipeGestureRecognizerDirection.Right; //default
            this.View.AddGestureRecognizer (gesture);
        }

        public void StartAlertOnSwipe ()
        {
            new UIAlertView ("Alert", "Swipe Recognized", null, "OK").Show ();
        }
    }
