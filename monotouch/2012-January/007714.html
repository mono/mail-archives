the Lion version of sqlite will not work on a phone, unless you have
that magic new ARM build of Lion (sqlite is a C app, so it's all
native code)

Works for me, at the moment, on iOS3,4,5.

I'm just calling:

 connection.CreateTable<Transaction>();

Transaction is just a class:

public class Transaction
        {
                public Transaction()
                {
                        Type = TransactionType.Cash;
                        Uploaded = false;
                        LastUploadDate = DateTime.MinValue;
                        InvoiceDate = DateTime.Now;
                        TransactionDate = DateTime.Now;
                }

                [PrimaryKey, AutoIncrement]
                public int Id { get; set;}
                public TransactionDirection Direction { get; set;}
                public string SupplierName { get; set;}
                public double Amount { get; set;}
                public string Description { get; set;}
                public string Reference { get; set;}
                public TransactionType Type { get; set;}
                public DateTime TransactionDate { get; set;}
                public string Notes { get; set; }
                public string ImageFile { get; set;}
                public bool Uploaded { get; set;}
                public DateTime LastUploadDate { get; set;}
                public DateTime InvoiceDate { get; set; }



        }
}
