<p class="MsoNormal" style="margin-bottom:12.0pt">here is the code for my DrawView.  This was heavily influenced by one of the samples in Dimitris'
<a href="http://www.amazon.com/iOS-Development-using-MonoTouch-Cookbook/dp/1849691460">
MonoTouch Cookbook</a>, so he deserves most of the credit.  <br>
<br>
In my controller (not shown) I have buttons for Clear, Cancel and Save that I display with the DrawView.  The DrawView has public methods for clearing the canvas and returning a UIImage of the signature/drawing.<br>
<br>
<br>
    public class DrawView : UIView<br>
    {<br>
<br>
        DrawViewController dvc;<br>
        <br>
        // clear the canvas<br>
        public void Clear ()<br>
        {<br>
            drawPath.Dispose ();<br>
            drawPath = new CGPath ();<br>
            fingerDraw = false;<br>
            SetNeedsDisplay ();<br>
<br>
        }<br>
        <br>
        // pass in a reference to the controller, although I never use it and could probably remove it<br>
        public DrawView (RectangleF frame, DrawViewController root) : base(frame)<br>
        {<br>
            dvc = root;<br>
            this.drawPath = new CGPath ();<br>
            this.BackgroundColor = UIColor.White;<br>
<br>
        }<br>
        <br>
        <br>
        private PointF touchLocation;<br>
        private PointF prevTouchLocation;<br>
        private CGPath drawPath;<br>
        private bool fingerDraw;<br>
        <br>
        public override void TouchesBegan (MonoTouch.Foundation.NSSet touches, UIEvent evt)<br>
        {<br>
            base.TouchesBegan (touches, evt);<br>
            <br>
            UITouch touch = touches.AnyObject as UITouch;<br>
            this.fingerDraw = true;<br>
            this.touchLocation = touch.LocationInView (this);<br>
            this.prevTouchLocation = touch.PreviousLocationInView (this);<br>
            this.SetNeedsDisplay ();<br>
            <br>
        }<br>
        <br>
        public override void TouchesMoved (MonoTouch.Foundation.NSSet touches, UIEvent evt)<br>
        {<br>
            base.TouchesMoved (touches, evt);<br>
            <br>
            UITouch touch = touches.AnyObject as UITouch;<br>
            this.touchLocation = touch.LocationInView (this);<br>
            this.prevTouchLocation = touch.PreviousLocationInView (this);<br>
            this.SetNeedsDisplay ();<br>
        }<br>
        <br>
        public UIImage GetDrawingImage ()<br>
        {<br>
            UIImage returnImg = null;<br>
            <br>
            UIGraphics.BeginImageContext (this.Bounds.Size);<br>
            <br>
            using (CGContext context = UIGraphics.GetCurrentContext()) {<br>
                context.SetStrokeColor (UIColor.Black.CGColor);<br>
                context.SetLineWidth (5f);<br>
                context.SetLineJoin (CGLineJoin.Round);<br>
                context.SetLineCap (CGLineCap.Round);<br>
                context.AddPath (this.drawPath);<br>
                context.DrawPath (CGPathDrawingMode.Stroke);<br>
                returnImg = UIGraphics.GetImageFromCurrentImageContext ();<br>
            }<br>
            <br>
            UIGraphics.EndImageContext ();<br>
        <br>
            return returnImg;<br>
        }<br>
        <br>
        <br>
        public override void Draw (RectangleF rect)<br>
        {<br>
            base.Draw (rect);<br>
            <br>
            if (this.fingerDraw) {<br>
                using (CGContext context = UIGraphics.GetCurrentContext()) {<br>
                    context.SetStrokeColor (UIColor.Black.CGColor);<br>
                    context.SetLineWidth (5f);<br>
                    context.SetLineJoin (CGLineJoin.Round);<br>
                    context.SetLineCap (CGLineCap.Round);<br>
                    this.drawPath.MoveToPoint (this.prevTouchLocation);<br>
                    this.drawPath.AddLineToPoint (this.touchLocation);<br>
                    context.AddPath (this.drawPath);<br>
                    context.DrawPath (CGPathDrawingMode.Stroke);<br>
                }<br>
            }    <br>
        }<br>
    }<br>
<br>
