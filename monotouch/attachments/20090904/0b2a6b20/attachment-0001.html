I found this method particularly useful:<br>
<br>
<span style='font-family:"Courier New"'>// Inspired by technique described in
section<br>
// &quot;Measuring Text Before Drawing&quot; of<br>
// &quot;Quartz 2D Programming Guide.&quot;<br>
// Note: only works for CGAffineTransform.MakeRotation(0)<br>
public static void ShowTextAtCenterPoint (<br>
&nbsp;&nbsp;&nbsp; this CGContext context, float centerX, float y, string text)<br>
{<br>
&nbsp;&nbsp;&nbsp; context.SaveState();<br>
&nbsp;&nbsp;&nbsp; context.SetTextDrawingMode (CGTextDrawingMode.Invisible);<br>
&nbsp;&nbsp;&nbsp; context.ShowTextAtPoint (centerX, y, text, text.Length);<br>
&nbsp;&nbsp;&nbsp; context.RestoreState();<br>
&nbsp;&nbsp;&nbsp; context.ShowTextAtPoint (<br>
&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp; centerX - (context.TextPosition.X -
centerX)/2,<br>
&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp; y,<br>
&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp; text,<br>
&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp; text.Length);<br>
}<br>
</span><br>
