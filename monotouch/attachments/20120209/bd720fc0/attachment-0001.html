I'm not sure 100% sure I solved this nicely, I've ended up using two<br>
Windows, something like this:<br>
<br>
        [Register ("AppDelegate")]<br>
        public class AppDelegate : UIApplicationDelegate<br>
        {<br>
                UIWindow _loginWindow, _mainWindow;<br>
<br>
                private UIViewController _loginViewController, _mainNavigationController;<br>
<br>
                bool _signedIn;<br>
                public override bool FinishedLaunching (UIApplication app, NSDictionary<br>
options)<br>
                {<br>
                        if (_signedIn) {<br>
                                ShowMainWindow ();<br>
                        } else<br>
                                ShowLoginWindow ();<br>
                        return true;<br>
                }<br>
<br>
                public void ShowLoginWindow ()<br>
                {<br>
                        if (_mainWindow != null) {<br>
                                _mainWindow.Hidden = true;<br>
                                _mainWindow.RemoveFromSuperview ();<br>
                                _mainWindow.Dispose ();<br>
                                _mainWindow = null;<br>
                        }<br>
                        _loginWindow = new UIWindow (UIScreen.MainScreen.Bounds);<br>
                        _loginViewController = new LoginViewController ();<br>
                        _loginWindow.RootViewController = _loginViewController;<br>
                        _loginWindow.MakeKeyAndVisible ();<br>
                }<br>
<br>
                public void ShowMainWindow ()<br>
                {<br>
                        if (_loginWindow != null) {<br>
                                _loginWindow.Hidden = true;<br>
                                _loginWindow.RemoveFromSuperview ();<br>
                                _loginWindow.Dispose ();<br>
                                _loginWindow = null;<br>
                        }<br>
                        _mainWindow = new UIWindow (UIScreen.MainScreen.Bounds);<br>
                        _mainNavigationController = new MainNavigationViewController ();<br>
                        _mainWindow.RootViewController = _mainNavigationController;<br>
                        _mainWindow.MakeKeyAndVisible ();<br>
                }<br>
        }<br>
<br>
<br>
When My LoginWindow has done its jobs it just calls ShowMainWindow smth like<br>
this:<br>
<br>
((AppDelegate)UIApplication.SharedApplication.Delegate).ShowMainWindow ();<br>
<br>
