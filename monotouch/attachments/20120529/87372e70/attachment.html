<div><span class="Apple-tab-span" style="white-space:pre">              </span></div><div>  Action<UITapGestureRecognizer> _onTouch;</div><div>  </div><div>  public SingleTapGestureRecognizer(Action<UITapGestureRecognizer> onTouch) : base() {</div>

<div>  public void SingleTapSelector(UITapGestureRecognizer sender) {</div><div>    _onTouch(sender);</div><div>  }</div><div><br></div><div>}</div><div><br></div><div>You can then do this on your UIPickerView:</div><div>
