I believe this works.  I'm not positive about the ref, but since you have a
pointer to a pointer I think that is what is needed.

[BaseType (typeof (NSObject))]
interface Test {
   [Export ("somefunction")]
   void SomeFunction();

   [Export ("somefunction:")]
   void SomeFunction(ref NSError err);

}
