I'm trying to use a UIScrollView for the first time - I have two pages of
landscape content I want the user to be able to switch between.

I'm hoping this is simply a Monday morning failure to fully activate my
brain, but my views are getting cut off at 768 pixels wide

anyone care to point out what I'm doing wrong?


    public class PagingController : UIViewController
    {
        UIScrollView _scroll;
        List<UIView> _pages;
        int _pagecount = 2;

        public PagingController ()
        {
        }

        public override void ViewDidLoad ()
        {
            base.ViewDidLoad ();


            // this view is only 768, should be 1024 (I'm in landscape)
            Console.WriteLine (this.View.Frame.Width);
            RectangleF scrollFrame = new RectangleF (0, 0, 1024, 768);
            scrollFrame.Width = scrollFrame.Width * 2;

            _pages = new List<UIView> ();

            _scroll = new UIScrollView{ Frame = View.Frame, PagingEnabled =
true, ContentSize = scrollFrame.Size, ShowsHorizontalScrollIndicator =
false };

            View.AddSubview (_scroll);

            for (int i = 0; i < _pagecount; i++) {

                UIView v = new UIView ();

                if (i == 0)
                    v.BackgroundColor = UIColor.Green;
                else
                    v.BackgroundColor = UIColor.Blue;

                v.Frame = new RectangleF (i * 1024, 0, 1024, 768);

                _pages.Add (v);

                _scroll.AddSubview (v);
            }
        }

        public override bool ShouldAutorotateToInterfaceOrientation
(UIInterfaceOrientation toInterfaceOrientation)
        {
            return (toInterfaceOrientation ==
UIInterfaceOrientation.LandscapeRight || toInterfaceOrientation ==
UIInterfaceOrientation.LandscapeLeft);
        }
    }
