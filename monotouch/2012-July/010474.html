I have it working just fine...  I did modify it though.

	/// <summary>
	/// Access the Camera.
	/// </summary>
	public static class Camera
	{
		static UIImagePickerController picker;
		static Action<NSDictionary> _callback;

		static void Init()
		{
			if(picker != null)
				return;

			picker = new UIImagePickerController();
			picker.AllowsEditing = false; // Don't allow editing for now as we'll
need to worry about saving edited images - also picker does only 320x320
cropping... ick
			picker.Delegate = new CameraDelegate();
		}

		class CameraDelegate : UIImagePickerControllerDelegate {
			public override void FinishedPickingMedia(UIImagePickerController picker,
NSDictionary info)
			{
				var cb = _callback;
				_callback = null;

				picker.DismissModalViewControllerAnimated(true);
				cb(info);
			}
		}

		public static bool HasCamera()
		{
			return
UIImagePickerController.IsSourceTypeAvailable(UIImagePickerControllerSourceType.Camera);
		}

		public static bool HasPhotoLibrary()
		{
			return
UIImagePickerController.IsSourceTypeAvailable(UIImagePickerControllerSourceType.PhotoLibrary);
		}

		private static readonly string[] Picture = new string[]{UTType.Image};
		private static readonly string[] Video = new string[]{UTType.Movie};
		private static readonly string[] Both = new string[]{UTType.Image,
UTType.Movie};

		/// <summary> Take a still picture </summary>
		public static void TakePhoto(UIViewController parent, Action<NSDictionary>
callback) {	GetMedia(parent, callback, Picture,
UIImagePickerControllerSourceType.Camera); }
		/// <summary> Take a video </summary>
		public static void TakeVideo(UIViewController parent, Action<NSDictionary>
callback) { GetMedia(parent, callback, Video,
UIImagePickerControllerSourceType.Camera); }
		/// <summary> Take a still picture or a video</summary>
		public static void TakeVideoOrPicture(UIViewController parent,
Action<NSDictionary> callback) {	GetMedia(parent, callback,
Both,UIImagePickerControllerSourceType.Camera); }

		/// <summary> Select a still picture </summary>
		public static void SelectPhoto(UIViewController parent,
Action<NSDictionary> callback) {	GetMedia(parent, callback, Picture,
UIImagePickerControllerSourceType.PhotoLibrary); }
		/// <summary> Select a video </summary>
		public static void SelectVideo(UIViewController parent,
Action<NSDictionary> callback) { GetMedia(parent, callback, Video,
UIImagePickerControllerSourceType.PhotoLibrary); }
		/// <summary> Select a still picture or a video</summary>
		public static void SelectVideoOrPicture(UIViewController parent,
Action<NSDictionary> callback) {	GetMedia(parent, callback, Both,
UIImagePickerControllerSourceType.PhotoLibrary); }

		private static void GetMedia(UIViewController parent, Action<NSDictionary>
callback, string[] requestedMediaTypes, UIImagePickerControllerSourceType
source)
		{
			Init();
			picker.SourceType = source;
			List<string> availableTypes = new
List<string>(UIImagePickerController.AvailableMediaTypes(source));
			picker.MediaTypes =
availableTypes.Intersect(requestedMediaTypes).ToArray();
			_callback = callback;
			parent.PresentModalViewController(picker, true);
		}
	}
