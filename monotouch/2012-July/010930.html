>                                                 return;
>
>                                         // Show the photo the user took
>                                         InvokeOnMainThread (delegate {
>                                                 UIImage img = UIImage.FromFile (t.Result.Path);
>                                                 UploadPhoto(img);
>                                         } );
>                                 }
>                                 );
