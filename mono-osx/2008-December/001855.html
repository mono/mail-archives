> 
> exec -a monodevelop $MONO $ASSEMBLY "$@"

> 
> OSX_VERSION=$(uname -r | cut -f1 -d.)
> 
> if [ $OSX_VERSION -lt 9 ]; then
>          $MONO $ASSEMBLY "$@"
> else
>          exec -a monodevelop $MONO $ASSEMBLY "$@"
> fi
