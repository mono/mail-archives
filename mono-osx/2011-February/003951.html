without much success so far.

                using OpenTK.Graphics.OpenGL;

                Console.WriteLine("MyOpenGLView.DrawRect() _version(-)");
                string _version = GL.GetString(StringName.Version);
                Console.WriteLine("MyOpenGLView.DrawRect() _version(+)=" +
