I use the following function to list all the processes running in my OS

 

public static void ListAllRunningProcesses()

        {

            // Get all processes on local machine.

            Process[] runningProcs = Process.GetProcessesByName("firefox");

            

            foreach (Process p in runningProcs)

            {

                

                string info = string.Format("-> PID: {0}\tName: {1}\t",

                    p.Id, p.ProcessName);

                Console.WriteLine(info);

            }

            Console.WriteLine("************************************\n");

        }
